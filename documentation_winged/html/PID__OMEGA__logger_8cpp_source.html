155 bool pid_omega_logger::logInitLogging() {
156  try {
157  // Create the directory path for the controller specific flight run logs
158  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "PID_OMEGA";
159 
160  // Check if the directory exits, and create it if it doesn't
161  if (!std::filesystem::exists(flight_run_controller_specific_directory))
162  {
163  std::filesystem::create_directories(flight_run_controller_specific_directory);
164  }
165 
166  // Generate the log file name with a timestamp
167  std::stringstream log_ss;
168  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
169  std::string log_filename = log_ss.str();
170 
171  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
172  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
173 
174  // Define a synchronous sink with a text ostream backend
175  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
176  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
177 
178  // Add a stream to the backend (in this case, a file stream)
179  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
180 
181  // Set the formatter for the sink
182  logdata_sink->set_formatter(
183  expr::stream
184  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
185  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
186  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
187  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
188  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
189  << expr::smessage // Log message
190  );
191 
192  // Add the sink to the logging core
193  logging::core::get()->add_sink(logdata_sink);
194 
195  // Set a filter for the sink
196  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
197 
198  logging::add_common_attributes(); // Add attributes like timestamp
199 
200  // Initilaize the logging headers
201  logInitHeaders();
202 
203  // Create the directory path for the controller specific flight run for parameters and gains used
204  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
205 
206  // Check if the direcotry exits, and create it if it doesn't
207  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
208  {
209  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
210  }
211 
212  // Copy the flight parameters over
213  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
214  std::stringstream params_target_ss;
215  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
216  std::string params_target_file = params_target_ss.str();
217 
218  if (!std::filesystem::exists(params_source_file)) {
219  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
220  }
221 
222  if (std::filesystem::exists(params_source_file)) {
223  std::filesystem::copy(params_source_file, params_target_file);
224  }
225 
226  // Copy the gains parameters over
227  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/qrbp/PID_OMEGA/gains_PID_OMEGA.json";
228  std::stringstream gains_target_ss;
229  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_PID_OMEGA.json" ;
230  std::string gains_target_file = gains_target_ss.str();
231 
232  if (!std::filesystem::exists(gains_source_file)) {
233  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
234  }
235 
236  if (std::filesystem::exists(gains_source_file)) {
237  std::filesystem::copy(gains_source_file, gains_target_file);
238  }
239 
240  // Return true if all successful
241  return true;
242 
243  } catch (const std::filesystem::filesystem_error& e) {
244  FLIGHTSTACK_ERROR("FIlesystem error:", e.what());
245  } catch (const std::exception& e) {
246  FLIGHTSTACK_ERROR("Exeption:", e.what());
247  }
248 
249  return false; // IF there are errors
250 }
