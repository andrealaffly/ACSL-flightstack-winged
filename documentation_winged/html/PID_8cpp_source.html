208 void pid::compute_translational_control_in_I()
209 {
210  // Compute the error in the states
211  cim.e_tran_pos << cim.x_tran_pos - cim.r_user;
212  cim.e_tran_vel << cim.x_tran_vel - cim.r_dot_user;
213 
214  // Compute the baseline control input
215  cim.mu_tran_baseline << MASS*( - cip.Kp_tran * cim.e_tran_pos
216  - cip.Kd_tran * cim.e_tran_vel
217  - cip.Ki_tran * csm.e_tran_pos_I
218  + cim.r_ddot_user );
219 
220  // Compute with the dynamic inversion
221  cim.mu_tran_I << cim.mu_tran_baseline - MASS * G * e3_basis;
222 }

224 void pid::compute_u1_eta_d()
225 {
226  // Compute the virtual forces in J
227  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
228 
229  // Compute u1 - Total Thrust
230  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
231  + pow(cim.mu_tran_J(1), 2)
232  + pow(cim.mu_tran_J(2), 2)
233  );
234 
235  // Compute the desired roll
236  double r1 = cim.mu_tran_J(1) / cim.u(0);
237  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
238 
239  // Compute the desired pitch
240  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
241 
242  // Compute the internal state
243  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
244  + B_filter_roll_ref * cim.eta_rot_d(0);
245 
246  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
247  + B_filter_pitch_ref * cim.eta_rot_d(1);
248 
249  cim.internal_state_roll_dot_d_filter << A_filter_roll_dot_ref * csm.state_roll_dot_d_filter
250  + B_filter_roll_dot_ref * cim.eta_rot_rate_d(0);
251 
252  cim.internal_state_pitch_dot_d_filter << A_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter
253  + B_filter_pitch_dot_ref * cim.eta_rot_rate_d(1);
254 
255  // Compute the desired pitch and roll rates
256  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
257  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
258 
259  // Compute the desired pitch and roll acceleration
260  cim.eta_rot_acceleration_d(0) = C_filter_roll_dot_ref * csm.state_roll_dot_d_filter;
261  cim.eta_rot_acceleration_d(1) = C_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter;
262 
263 }

265 void pid::compute_rotational_control()
266 {
267  // Compute the error in the rotational states
268  cim.e_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
269  cim.e_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
270  cim.e_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
271 
272  cim.e_eta_rate << cim.eta_rot_rate - cim.eta_rot_rate_d;
273 
274  // Compute the baseline control input
275  cim.tau_rot_baseline << inertia_matrix_q * ( - cip.Kp_rot * cim.e_eta
276  - cip.Kd_rot * cim.e_eta_rate
277  - cip.Ki_rot * csm.e_eta_I
278  + cim.eta_rot_acceleration_d
279  );
280 
281  // Compute with dynamic inversion
282  cim.tau_rot << cim.tau_rot_baseline + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot);
283 
284  // Assign the control inputs
285  cim.u(1) = cim.tau_rot(0);
286  cim.u(2) = cim.tau_rot(1);
287  cim.u(3) = cim.tau_rot(2);
288 
289 }

311 void pid::run(const double time_step_rk4_) {
312 
313  // Process the dynamics --------------------------------------------------------
314  // 1. Compute the aerodynamics
315 
316  // 2. Compute the translational control input
317  compute_translational_control_in_I();
318 
319  // 3. Compute the thrust needed and the desired angles
320  compute_u1_eta_d();
321 
322  // 4. Compute the rotational control input
323  compute_rotational_control();
324 
325  // 5. Compute the normalized thrust - Final Step
326  compute_normalized_thrusts();
327 
328  // 6. Do the integration
329  rk4.do_step(boost::bind(&pid::model, this, bph::_1, bph::_2, bph::_3),
330  y, cim.t, time_step_rk4_);
331 
332  // Capture the time after the execution of the controller
333  cim.alg_end_time = std::chrono::high_resolution_clock::now();
334 
335  // Calculate the duration of the execution
336  cim.alg_duration = std::chrono::duration_cast<std::chrono::microseconds>(
337  cim.alg_end_time - cim.alg_start_time).count();
338 
339  // 7. Log the Data after all the calculataions
340  logger.logLogData();
341 
342  // optional debug
343  debug2terminal();
344 }
