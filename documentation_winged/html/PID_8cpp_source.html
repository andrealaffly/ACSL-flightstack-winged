198 void pid::compute_translational_control_in_I()
199 {
200  // Compute the error in the states
201  cim.e_tran_pos << cim.x_tran_pos - cim.r_user;
202  cim.e_tran_vel << cim.x_tran_vel - cim.r_dot_user;
203 
204  // Compute the baseline control input
205  cim.mu_tran_baseline << MASS*( - cip.Kp_tran * cim.e_tran_pos
206  - cip.Kd_tran * cim.e_tran_vel
207  - cip.Ki_tran * csm.e_tran_pos_I
208  + cim.r_ddot_user );
209 
210  // Compute with the dynamic inversion
211  cim.mu_tran_I << cim.mu_tran_baseline - MASS * G * e3_basis;
212 }

214 void pid::compute_u1_eta_d()
215 {
216  // Compute the virtual forces in J
217  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
218 
219  // Compute u1 - Total Thrust
220  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
221  + pow(cim.mu_tran_J(1), 2)
222  + pow(cim.mu_tran_J(2), 2)
223  );
224 
225  // Compute the desired roll
226  double r1 = cim.mu_tran_J(1) / cim.u(0);
227  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
228 
229  // Compute the desired pitch
230  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
231 
232  // Compute the internal state
233  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
234  + B_filter_roll_ref * cim.eta_rot_d(0);
235 
236  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
237  + B_filter_pitch_ref * cim.eta_rot_d(1);
238 
239  cim.internal_state_roll_dot_d_filter << A_filter_roll_dot_ref * csm.state_roll_dot_d_filter
240  + B_filter_roll_dot_ref * cim.eta_rot_rate_d(0);
241 
242  cim.internal_state_pitch_dot_d_filter << A_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter
243  + B_filter_pitch_dot_ref * cim.eta_rot_rate_d(1);
244 
245  // Compute the desired pitch and roll rates
246  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
247  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
248 
249  // Compute the desired pitch and roll acceleration
250  cim.eta_rot_acceleration_d(0) = C_filter_roll_dot_ref * csm.state_roll_dot_d_filter;
251  cim.eta_rot_acceleration_d(1) = C_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter;
252 
253 }

255 void pid::compute_rotational_control()
256 {
257  // Compute the error in the rotational states
258  cim.e_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
259  cim.e_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
260  cim.e_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
261 
262  cim.e_eta_rate << cim.eta_rot_rate - cim.eta_rot_rate_d;
263 
264  // Compute the baseline control input
265  cim.tau_rot_baseline << inertia_matrix_q * ( - cip.Kp_rot * cim.e_eta
266  - cip.Kd_rot * cim.e_eta_rate
267  - cip.Ki_rot * csm.e_eta_I
268  + cim.eta_rot_acceleration_d
269  );
270 
271  // Compute with dynamic inversion
272  cim.tau_rot << cim.tau_rot_baseline + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot);
273 
274  // Assign the control inputs
275  cim.u(1) = cim.tau_rot(0);
276  cim.u(2) = cim.tau_rot(1);
277  cim.u(3) = cim.tau_rot(2);
278 
279 }

301 void pid::run(const double time_step_rk4_) {
302 
303  // Process the dynamics --------------------------------------------------------
304  // 1. Compute the aerodynamics
305 
306  // 2. Compute the translational control input
307  compute_translational_control_in_I();
308 
309  // 3. Compute the thrust needed and the desired angles
310  compute_u1_eta_d();
311 
312  // 4. Compute the rotational control input
313  compute_rotational_control();
314 
315  // 5. Compute the normalized thrust - Final Step
316  compute_normalized_thrusts();
317 
318  // 6. Do the integration
319  rk4.do_step(boost::bind(&pid::model, this, bph::_1, bph::_2, bph::_3),
320  y, cim.t, time_step_rk4_);
321 
322  // Capture the time after the execution of the controller
323  cim.alg_end_time = std::chrono::high_resolution_clock::now();
324 
325  // Calculate the duration of the execution
326  cim.alg_duration = std::chrono::duration_cast<std::chrono::microseconds>(
327  cim.alg_end_time - cim.alg_start_time).count();
328 
329  // 7. Log the Data after all the calculataions
330  logger.logLogData();
331 
332  // optional debug
333  debug2terminal();
334 }
