130  cim.q.z() = q3;
131 
132  // 2. Get the reference trajectory ---------------------------------------------
133  ud.updateUserDefinedTrajectory(time);
134  cim.r_user = ud.getUserDefinedPosition();
135  cim.r_dot_user = ud.getUserDefinedVelocity();
136  cim.r_ddot_user = ud.getUserDefinedAcceleration();
137 
138  // Set the desired yaw, yaw rate and yaw acceleration
139  cim.eta_rot_d(2) = ud.getUserDefinedYaw();
140  cim.eta_rot_rate_d(2) = ud.getUserDefinedYawRate();
141  cim.eta_rot_acceleration_d(2) = ud.getUserDefinedYawRateDot();
142 
143  // 3. Capture the time before the execution of the controller ------------------
144  cim.alg_start_time = std::chrono::high_resolution_clock::now();
145 
146  // 4. Compute the rotation matrices and the jacobian Inverse -------------------
147  cim.R_J_I = rotationMatrix321LocalToGlobal(cim.eta_rot(0), cim.eta_rot(1), cim.eta_rot(2));

152  cim.eta_rot_rate = cim.Jacobian_inv*cim.omega_rot;

155  assign_from_rk4();

183 void pid::compute_translational_control_in_I()
184 {
185  // Compute the error in the states
186  cim.e_tran_pos << cim.x_tran_pos - cim.r_user;
187  cim.e_tran_vel << cim.x_tran_vel - cim.r_dot_user;
188 
189  // Compute the baseline control input
190  cim.mu_tran_baseline << MASS*( - cip.Kp_tran * cim.e_tran_pos
191  - cip.Kd_tran * cim.e_tran_vel
192  - cip.Ki_tran * csm.e_tran_pos_I
193  + cim.r_ddot_user );
194 
195  // Compute with the dynamic inversion
196  cim.mu_tran_I << cim.mu_tran_baseline - MASS * G * e3_basis;
197 }

199 void pid::compute_u1_eta_d()
200 {
201  // Compute the virtual forces in J
202  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
203 
204  // Compute u1 - Total Thrust
205  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
206  + pow(cim.mu_tran_J(1), 2)
207  + pow(cim.mu_tran_J(2), 2)
208  );
209 
210  // Compute the desired roll
211  double r1 = cim.mu_tran_J(1) / cim.u(0);
212  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
213 
214  // Compute the desired pitch
215  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
216 
217  // Compute the internal state
218  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
219  + B_filter_roll_ref * cim.eta_rot_d(0);
220 
221  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
222  + B_filter_pitch_ref * cim.eta_rot_d(1);
223 
224  cim.internal_state_roll_dot_d_filter << A_filter_roll_dot_ref * csm.state_roll_dot_d_filter
225  + B_filter_roll_dot_ref * cim.eta_rot_rate_d(0);
226 
227  cim.internal_state_pitch_dot_d_filter << A_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter
228  + B_filter_pitch_dot_ref * cim.eta_rot_rate_d(1);
229 
230  // Compute the desired pitch and roll rates
231  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
232  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
233 
234  // Compute the desired pitch and roll acceleration
235  cim.eta_rot_acceleration_d(0) = C_filter_roll_dot_ref * csm.state_roll_dot_d_filter;
236  cim.eta_rot_acceleration_d(1) = C_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter;
237

241 void pid::compute_rotational_control()
242 {
243  // Compute the error in the rotational states
244  cim.e_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
245  cim.e_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
246  cim.e_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
247 
248  cim.e_eta_rate << cim.eta_rot_rate - cim.eta_rot_rate_d;
249 
250  // Compute the baseline control input
251  cim.tau_rot_baseline << inertia_matrix_q * ( - cip.Kp_rot * cim.e_eta
252  - cip.Kd_rot * cim.e_eta_rate
253  - cip.Ki_rot * csm.e_eta_I
254  + cim.eta_rot_acceleration_d
255  );
256 
257  // Compute with dynamic inversion
258  cim.tau_rot << cim.tau_rot_baseline + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot);
259 
260  // Assign the control inputs
261  cim.u(1) = cim.tau_rot(0);
262  cim.u(2) = cim.tau_rot(1);
263  cim.u(3) = cim.tau_rot(2);
264 
265 }

268 {
269  // Compute the individual thrusts in Newtons
270  cim.Thrust << mixer_matrix_qrbp * cim.u;

287 void pid::run(const double time_step_rk4_) {
288 
289  // Process the dynamics --------------------------------------------------------
290  // 1. Compute the translational control input
291  compute_translational_control_in_I();
292 
293  // 2. Compute the thrust needed and the desired angles
294  compute_u1_eta_d();
295 
296  // 4. Compute the rotational control input
297  compute_rotational_control();
298 
299  // 5. Compute the normalized thrust - Final Step
300  compute_normalized_thrusts();
301 
302  // 6. Do the integration
303  rk4.do_step(boost::bind(&pid::model, this, bph::_1, bph::_2, bph::_3),
304  y, cim.t, time_step_rk4_);
305 
306  // Capture the time after the execution of the controller
307  cim.alg_end_time = std::chrono::high_resolution_clock::now();
308 
309  // Calculate the duration of the execution
310  cim.alg_duration = std::chrono::duration_cast<std::chrono::microseconds>(
311  cim.alg_end_time - cim.alg_start_time).count();
312 
313  // 7. Log the Data after all the calculataions
314  logger.logLogData();
315 
316  // optional debug
317  debug2terminal();
318 }
