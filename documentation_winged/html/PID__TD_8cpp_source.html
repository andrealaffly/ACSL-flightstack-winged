138  cim.q.z() = q3;
139 
140  // 2. Get the reference trajectory ---------------------------------------------
141  ud.updateUserDefinedTrajectory(time);
142  cim.r_user = ud.getUserDefinedPosition();
143  cim.r_dot_user = ud.getUserDefinedVelocity();
144  cim.r_ddot_user = ud.getUserDefinedAcceleration();
145 
146  // Set the desired yaw, yaw rate and yaw acceleration
147  cim.eta_rot_d(2) = ud.getUserDefinedYaw();
148  cim.eta_rot_rate_d(2) = ud.getUserDefinedYawRate();

153

155  cim.R_J_I = rotationMatrix321LocalToGlobal(cim.eta_rot(0), cim.eta_rot(1), cim.eta_rot(2));
156  cim.R_I_J = rotationMatrix321GlobalToLocal(cim.eta_rot(0), cim.eta_rot(1), cim.eta_rot(2));

160  cim.eta_rot_rate = cim.Jacobian_inv*cim.omega_rot;

163  assign_from_rk4();

191 void pid::compute_translational_control_in_I()
192 {
193  // Compute the error in the states
194  cim.e_tran_pos << cim.x_tran_pos - cim.r_user;
195  cim.e_tran_vel << cim.x_tran_vel - cim.r_dot_user;
196 
197  // Compute the baseline control input
198  cim.mu_tran_baseline << MASS*( - cip.Kp_tran * cim.e_tran_pos
199  - cip.Kd_tran * cim.e_tran_vel
200  - cip.Ki_tran * csm.e_tran_pos_I
201  + cim.r_ddot_user );
202 
203  // Compute with the dynamic inversion
204  cim.mu_tran_I << cim.mu_tran_baseline - MASS * G * e3_basis;
205 }

207 void pid::compute_u1_eta_d()
208 {
209  // Compute the virtual forces in J
210  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
211 
212  // Compute u1 - Total Thrust
213  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
214  + pow(cim.mu_tran_J(1), 2)
215  + pow(cim.mu_tran_J(2), 2)
216  );
217 
218  // Compute the desired roll
219  double r1 = cim.mu_tran_J(1) / cim.u(0);
220  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
221 
222  // Compute the desired pitch
223  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
224 
225  // Compute the internal state
226  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
227  + B_filter_roll_ref * cim.eta_rot_d(0);
228 
229  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
230  + B_filter_pitch_ref * cim.eta_rot_d(1);
231 
232  cim.internal_state_roll_dot_d_filter << A_filter_roll_dot_ref * csm.state_roll_dot_d_filter
233  + B_filter_roll_dot_ref * cim.eta_rot_rate_d(0);
234 
235  cim.internal_state_pitch_dot_d_filter << A_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter
236  + B_filter_pitch_dot_ref * cim.eta_rot_rate_d(1);
237 
238  // Compute the desired pitch and roll rates
239  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
240  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
241 
242  // Compute the desired pitch and roll acceleration
243  cim.eta_rot_acceleration_d(0) = C_filter_roll_dot_ref * csm.state_roll_dot_d_filter;
244  cim.eta_rot_acceleration_d(1) = C_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter;
245

249 void pid::compute_rotational_control()
250 {
251  // Compute the error in the rotational states
252  cim.e_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
253  cim.e_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
254  cim.e_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
255 
256  cim.e_eta_rate << cim.eta_rot_rate - cim.eta_rot_rate_d;
257 
258  // Compute the baseline control input
259  cim.tau_rot_baseline << inertia_matrix_q * ( - cip.Kp_rot * cim.e_eta
260  - cip.Kd_rot * cim.e_eta_rate
261  - cip.Ki_rot * csm.e_eta_I
262  + cim.eta_rot_acceleration_d
263  );
264 
265  // Compute with dynamic inversion
266  cim.tau_rot << cim.tau_rot_baseline + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot);
267 
268  // Assign the control inputs
269  cim.u(1) = cim.tau_rot(0);
270  cim.u(2) = cim.tau_rot(1);
271  cim.u(3) = cim.tau_rot(2);
272 
273 }

295 void pid::run(const double time_step_rk4_) {
296 
297  // Process the dynamics --------------------------------------------------------
298  // 1. Compute the translational control input
299  compute_translational_control_in_I();
300 
301  // 2. Compute the thrust needed and the desired angles
302  compute_u1_eta_d();
303 
304  // 4. Compute the rotational control input
305  compute_rotational_control();
306 
307  // 5. Compute the normalized thrust - Final Step
308  compute_normalized_thrusts();
309 
310  // 6. Do the integration
311  rk4.do_step(boost::bind(&pid::model, this, bph::_1, bph::_2, bph::_3),
312  y, cim.t, time_step_rk4_);
313 
314  // Capture the time after the execution of the controller
315  cim.alg_end_time = std::chrono::high_resolution_clock::now();
316 
317  // Calculate the duration of the execution
318  cim.alg_duration = std::chrono::duration_cast<std::chrono::microseconds>(
319  cim.alg_end_time - cim.alg_start_time).count();
320 
321  // 7. Log the Data after all the calculataions
322  logger.logLogData();
323 
324  // optional debug
325  debug2terminal();
