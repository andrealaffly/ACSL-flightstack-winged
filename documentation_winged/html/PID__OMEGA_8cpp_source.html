112 void pid_omega::update(double time,
113  double x,
114  double y,
115  double z,
116  double vx,
117  double vy,
118  double vz,
119  double q0,
120  double q1,
121  double q2,
122  double q3,
123  double roll,
124  double pitch,
125  double yaw,
126  double w_x,
127  double w_y,
128  double w_z)
129 {
130  // 1. Assign all the states ----------------------------------------------------
131  cim.t = time;
132 
133  // Assign the Translational States
134  cim.x_tran_pos << x,y,z;
135  cim.x_tran_vel << vx, vy, vz;
136 
137  // Assign the Rotational States
138  cim.eta_rot << roll, pitch, yaw;
139  cim.omega_rot << w_x, w_y, w_z;
140 
141  // Assign the quaternions
142  cim.q.w() = q0;
143  cim.q.z() = q1;
144  cim.q.y() = q2;
145  cim.q.z() = q3;
146 
147  // 2. Get the reference trajectory ---------------------------------------------
148  ud.updateUserDefinedTrajectory(time);
149  cim.r_user = ud.getUserDefinedPosition();
150  cim.r_dot_user = ud.getUserDefinedVelocity();
151  cim.r_ddot_user = ud.getUserDefinedAcceleration();
152 
153  // Set the desired yaw and yaw_rate; we don't need the acceleration as
154  // we are going to compute it from the filter.
155  cim.eta_rot_d(2) = ud.getUserDefinedYaw();
156  cim.eta_rot_rate_d(2) = ud.getUserDefinedYawRate();
157 
158  // 3. Capture the time before the execution of the controller ------------------
159  cim.alg_start_time = std::chrono::high_resolution_clock::now();
160 
161  // 4. Compute the rotation matrices --------------------------------------------
162  cim.R_I_J = rotationMatrix321GlobalToLocal(cim.eta_rot(0), cim.eta_rot(1), cim.eta_rot(2));
163  cim.R_J_I = rotationMatrix321LocalToGlobal(cim.eta_rot(0), cim.eta_rot(1), cim.eta_rot(2));
164 
165  // 5. Compute the body velocities in the quadcopter frame from the inertial velocities
166  cim.x_tran_vel_J = cim.R_I_J * cim.x_tran_vel;
167 
168  // 6. Compute the square of the norm of velocity in the body frame - it is the same for both quadcopter and biplane
169  // amd compute the aerodynamic angles
170  cim.aero.states = computeAeroStatesNACA0012Quad(cim.x_tran_vel_J(0), cim.x_tran_vel_J(1), cim.x_tran_vel_J(2),
171  cim.omega_rot(0), cim.omega_rot(1), cim.omega_rot(2));
172 
173  // 7. Compute the Direction Cosine Matrix in the quadcopter frame at the center of gravity
174  cim.R_W_J = aeroDCMQuadNED(cim.aero.states.alpha, cim.aero.states.beta);
175 
176  // 8. Compute the aerodynamic coefficeints
177  cim.aero.coeff = compute_aero_coefficients(cim.aero.states);
178 
179  // 10. Assign the values from the integrator ------------------------------------
180  assign_from_rk4();
181 }

211 // Function to compute the outerloop control in I
212 void pid_omega::compute_translational_control_in_I()
213 {
214  // Compute the error in the states
215  cim.e_tran_pos << cim.x_tran_pos - cim.r_user;
216  cim.e_tran_vel << cim.x_tran_vel - cim.r_dot_user;
217 
218  // Compute the baseline control input
219  cim.mu_tran_baseline << MASS*( - cip.Kp_tran * cim.e_tran_pos
220  - cip.Kd_tran * cim.e_tran_vel
221  - cip.Ki_tran * csm.e_tran_pos_I
222  + cim.r_ddot_user );
223 
224  // Compute with the dynamic inversion without aerodynamics
225  cim.mu_tran_I << cim.mu_tran_baseline - MASS * G * e3_basis;
226 }

228 // Compute the orientation commands and the desired total thrust
229 void pid_omega::compute_u1_eta_d()
230 {
231  // Compute the virtual forces in J
232  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
233 
234  // Compute u1 - Total Thrust
235  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
236  + pow(cim.mu_tran_J(1), 2)
237  + pow(cim.mu_tran_J(2), 2)
238  );
239 
240  // Compute the desired roll
241  double r1 = cim.mu_tran_J(1) / cim.u(0);
242  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
243 
244  // Compute the desired pitch
245  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
246 
247  // Compute the internal state for angular rates
248  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
249  + B_filter_roll_ref * cim.eta_rot_d(0);
250 
251  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
252  + B_filter_pitch_ref * cim.eta_rot_d(1);
253 
254  // Compute the desired pitch and roll rates
255  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
256  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
257 
258  // Compute the Jacobian with the current orientation
259  cim.Jacobian = jacobianMatrix(cim.eta_rot(0), cim.eta_rot(1));
260 
261  // Compute the desired angular velocities
262  cim.omega_rot_d << cim.Jacobian * cim.eta_rot_rate_d;
263 
264  // Compute the internal state for angular accelration
265  cim.internal_state_omega_x_d_filter << A_filter_roll_dot_ref * csm.state_omega_x_d_filter
266  + B_filter_roll_dot_ref * cim.omega_rot_d(0);
267 
268  cim.internal_state_omega_y_d_filter << A_filter_pitch_dot_ref * csm.state_omega_y_d_filter
269  + B_filter_pitch_dot_ref * cim.omega_rot_d(1);
270 
271  cim.internal_state_omega_z_d_filter << A_filter_yaw_dot_ref * csm.state_omega_z_d_filter
272  + B_filter_yaw_dot_ref * cim.omega_rot_d(2);
273 
274  // Compute the desired angular acceleration
275  cim.alpha_rot_d(0) = C_filter_roll_dot_ref * csm.state_omega_x_d_filter;
276  cim.alpha_rot_d(1) = C_filter_pitch_dot_ref * csm.state_omega_y_d_filter;
277  cim.alpha_rot_d(2) = C_filter_yaw_dot_ref * csm.state_omega_z_d_filter;
278 }

281 // Compute the rotational control
282 void pid_omega::compute_rotational_control()
283 {
284  // Compute the error in the rotational states
285  cim.e_rot_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
286  cim.e_rot_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
287  cim.e_rot_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
288 
289  cim.e_rot_omega << cim.omega_rot - cim.omega_rot_d;
290 
291  // Compute the baseline control input
292  cim.tau_rot_baseline << inertia_matrix_q * ( - cip.Kp_rot * cim.e_rot_eta
293  - cip.Kd_rot * cim.e_rot_omega
294  - cip.Ki_rot * csm.e_rot_eta_I
295  + cim.alpha_rot_d);
296 
297  // Compute with dynamic inversion without aerodynamics
298  cim.tau_rot << cim.tau_rot_baseline
299  + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot);
300 
301  // Assign the control inputs
302  cim.u(1) = cim.tau_rot(0);
303  cim.u(2) = cim.tau_rot(1);
304  cim.u(3) = cim.tau_rot(2);
305 }

307 // Function to compute the normalized thrusts
308 void pid_omega::compute_normalized_thrusts()
309 {
310  // Compute the individual thrusts in Newtons
311  cim.Thrust << mixer_matrix_qrbp * cim.u;
312 
313  // Saturate each element of the Thrust vector between MIN_THRUST and MAX_THRUST
314  cim.Sat_Thrust = (cim.Thrust.cwiseMin(MAX_THRUST).cwiseMax(MIN_THRUST));
315 
316  // Compute the final control inputs
317  control_input(0) = (float) evaluatePolynomial(thrust_polynomial_coeff_qrbp, cim.Sat_Thrust(0));
318  control_input(1) = (float) evaluatePolynomial(thrust_polynomial_coeff_qrbp, cim.Sat_Thrust(1));
319  control_input(2) = (float) evaluatePolynomial(thrust_polynomial_coeff_qrbp, cim.Sat_Thrust(2));
320  control_input(3) = (float) evaluatePolynomial(thrust_polynomial_coeff_qrbp, cim.Sat_Thrust(3));

329 // Function that is called in control.cpp
330 void pid_omega::run(const double time_step_rk4_) {
331 
332  // Process the dynamics --------------------------------------------------------
333  // 1. Compute the aerodynamics
334  compute_aero_forces_moments(cim.aero.states, cim.aero.coeff, cim.R_J_I, cim.R_W_J);
335 
336  // 2. Compute the translational control input
337  compute_translational_control_in_I();
338 
339  // 3. Compute the thrust needed and the desired angles
340  compute_u1_eta_d();
341 
342  // 4. Compute the rotational control input
343  compute_rotational_control();
344 
345  // 5. Compute the normalized thrust - Final Step
346  compute_normalized_thrusts();
347 
348  // 6. Do the integration
349  rk4.do_step(boost::bind(&pid_omega::model, this, bph::_1, bph::_2, bph::_3),
350  y, cim.t, time_step_rk4_);
351 
352  // Capture the time after the execution of the controller
353  cim.alg_end_time = std::chrono::high_resolution_clock::now();
354 
355  // Calculate the duration of the execution
356  cim.alg_duration = std::chrono::duration_cast<std::chrono::microseconds>(
357  cim.alg_end_time - cim.alg_start_time).count();
358 
359  // 7. Log the Data after all the calculataions
360  logger.logLogData();
361 
362  // optional debug
363  debug2terminal();
364 }
