102 bool UdpReceiverNode::logInitLogging() {
103  try {
104  // Generate the log file name
105  std::stringstream log_ss;
106  log_ss << flight_run_log_directory << "/mocap_log" << ".log";
107  std::string mocap_log_filename = log_ss.str();
108 
109  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
110  logger_mocapdata.add_attribute("Tag", attrs::constant< std::string >("MocapTag"));
111 
112  // Define a synchronous sink with a text ostream backend
113  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
114  boost::shared_ptr<text_sink> mocap_sink = boost::make_shared<text_sink>();
115 
116  // Add a stream to the backend (in this case, a file stream)
117  mocap_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(mocap_log_filename));
118 
119  // Set the formatter for the sink
120  mocap_sink->set_formatter(
121  expr::stream
122  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
123  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
124  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
125  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
126  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
127  << expr::smessage // Log message
128  );
129 
130  // Add the sink to the logging core
131  logging::core::get()->add_sink(mocap_sink);
132 
133  // Set a filter for the sink
134  mocap_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "MocapTag");
135 
136  logging::add_common_attributes(); // Add attributes like timestamp
137 
138  // Initilaize the logging headers
139  logInitHeaders();
140 
141  // Return true if all successful
142  return true;
143 
144  } catch (const std::filesystem::filesystem_error& e) {
145  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
146  } catch (const std::exception& e) {
147  FLIGHTSTACK_ERROR("Exception: ", e.what());
148  }
149 
150  return false; // IF there are errors
151 }

217 LNI::CallbackReturn UdpReceiverNode::on_configure(const lc::State & state)
218 {
219  (void)state;
220 
221  // Check if m_udp_driver is not null
222  if (m_udp_driver) {
223  RCLCPP_INFO(get_logger(), "UDP driver is initialized.");
224  }
225 
226  // Create a Sensor Data QoS profile - 285-330Hz is data frequency of MOCAP. Message every 3.51-3.03ms.
227  // Therefore, we want to discard old messages in the queue.
228  rclcpp::QoS qos_settings = rclcpp::SensorDataQoS(); // Use default Sensor Data QoS profile
229  qos_settings.keep_last(1); // Keep only the last message
230 
231  // Create publisher for mocap to the pixhawk visual odometry topic
232  mocap_publisher_ = this->create_publisher<px4_msgs::msg::VehicleOdometry>(
233  "/fmu/in/vehicle_visual_odometry", qos_settings);
234 
235  try {
236 
237  m_udp_driver->init_receiver(m_ip, m_port);
238  m_udp_driver->receiver()->open();
239  m_udp_driver->receiver()->bind();
240  m_udp_driver->receiver()->asyncReceive(
241  std::bind(&UdpReceiverNode::receiver_callback, this, std::placeholders::_1));
242  } catch (const std::exception & ex) {
243  RCLCPP_ERROR(
244  get_logger(), "Error creating UDP receiver: %s:%i - %s",
245  m_ip.c_str(), m_port, ex.what());
246  return LNI::CallbackReturn::FAILURE;
247  }
248 
249  RCLCPP_DEBUG(get_logger(), "UDP receiver successfully configured.");
250 
251  return LNI::CallbackReturn::SUCCESS;
252 }

324 void UdpReceiverNode::receiver_callback(const std::vector<uint8_t> & buffer)
325 {
326  // Convert the buffer to a string
327  std::string buffer_str(buffer.begin(), buffer.end());
328 
329  // Check if the message format is valid
330  if (buffer_str.size() < 2 || buffer_str[0] != 'R' || buffer_str[1] != ',') {
331  FLIGHTSTACK_ERROR("Invalid message format!");
332  return; // Exit the function or handle the error accordingly
333  }
334 
335 
336  // Parse the data from the string
337  std::istringstream iss(buffer_str.substr(2)); // Skip "R," prefix
338  char comma;
339  iss >> mc_im.x >> comma >> mc_im.y >> comma >> mc_im.z >> comma // Read in the position
340  >> mc_im.q0 >> comma >> mc_im.q1 >> comma >> mc_im.q2 >> comma >> mc_im.q3 >> comma // Read in the quaternion
341  >> mc_im.vx >> comma >> mc_im.vy >> comma >> mc_im.vz >> comma // Read in the velocities
342  >> mc_im.rollspeed >> comma >> mc_im.pitchspeed >> comma >> mc_im.yawspeed; // Read in the angular speeds
343 
344  // Get the controller time
345  mc_im.control_time = vehicle_ptr->get_controltime();
346 
347  // Debug Mocap parsed data to terminal
348  // debugMocapData2screen();
349 
350  // Initilize the Mocap odometry message to publish to the pixhawk
351  px4_msgs::msg::VehicleOdometry out;
352 
353  // \brief: https://github.com/hpaul360/mocap_to_px4/blob/main/mocap_to_px4/converter_member_function.py
354  // Package the message
355  out.timestamp = 0; // Timestamp is automatically set inside PX4
356  out.timestamp_sample = 0; // Timestamp is automatically set inside PX4
357 
358  // Position frame is set to NED in the VICON code.
359  out.position[0] = (float) mc_im.x;
360  out.position[1] = (float) mc_im.y;
361  out.position[2] = (float) mc_im.z;
362 
363  // Orientation frame is set to NED in the VICON code.
364  out.q[0] = (float) mc_im.q0;
365  out.q[1] = (float) mc_im.q1;
366  out.q[2] = (float) mc_im.q2;
367  out.q[3] = (float) mc_im.q3;
368  out.pose_frame = POSE_FRAME_NED;
369 
370  // Velocity frame is set to NED in the VICON code.
371  // The velocity calculations in the VICON code is not great, once updated - fuse.
372  out.velocity[0] = (float) mc_im.vx;
373  out.velocity[1] = (float) mc_im.vy;
374  out.velocity[2] = (float) mc_im.vz;
375  out.velocity_frame = VELOCITY_FRAME_NED;
376 
377  // Angular speed is set to 0. We need to pass the velocites to EKF2 in the FRD Body-Fixed
378  // Frame. We do not have the data in the VICON code therefore we set it to 0.
379  out.angular_velocity[0] = (float) 0.0;
380  out.angular_velocity[1] = (float) 0.0;
381  out.angular_velocity[2] = (float) 0.0;
382 
383  // Other Parameters
384  out.position_variance = {0,0,0};
385  out.orientation_variance = {0,0,0};
386  out.velocity_variance = {0,0,0};
387 
388  // Publish the message
389  mocap_publisher_->publish(out);
390 
391  // Log the data
392  logLogData();
393 }
