91 bool UdpReceiverNode::logInitLogging() {
92  try {
93  // Generate the log file name
94  std::stringstream log_ss;
95  log_ss << flight_run_log_directory << "/mocap_log" << ".log";
96  std::string mocap_log_filename = log_ss.str();
97 
98  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
99  logger_mocapdata.add_attribute("Tag", attrs::constant< std::string >("MocapTag"));
100 
101  // Define a synchronous sink with a text ostream backend
102  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
103  boost::shared_ptr<text_sink> mocap_sink = boost::make_shared<text_sink>();
104 
105  // Add a stream to the backend (in this case, a file stream)
106  mocap_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(mocap_log_filename));
107 
108  // Set the formatter for the sink
109  mocap_sink->set_formatter(
110  expr::stream
111  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
112  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
113  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
114  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
115  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
116  << expr::smessage // Log message
117  );
118 
119  // Add the sink to the logging core
120  logging::core::get()->add_sink(mocap_sink);
121 
122  // Set a filter for the sink
123  mocap_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "MocapTag");
124 
125  logging::add_common_attributes(); // Add attributes like timestamp
126 
127  // Initilaize the logging headers
128  logInitHeaders();
129 
130  // Return true if all successful
131  return true;
132 
133  } catch (const std::filesystem::filesystem_error& e) {
134  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
135  } catch (const std::exception& e) {
136  FLIGHTSTACK_ERROR("Exception: ", e.what());
137  }
138 
139  return false; // IF there are errors
140 }

206 LNI::CallbackReturn UdpReceiverNode::on_configure(const lc::State & state)
207 {
208  (void)state;
209 
210  // Check if m_udp_driver is not null
211  if (m_udp_driver) {
212  RCLCPP_INFO(get_logger(), "UDP driver is initialized.");
213  }
214 
215  // Create a Sensor Data QoS profile - 285-330Hz is data frequency of MOCAP. Message every 3.51-3.03ms.
216  // Therefore, we want to discard old messages in the queue.
217  rclcpp::QoS qos_settings = rclcpp::SensorDataQoS(); // Use default Sensor Data QoS profile
218  qos_settings.keep_last(1); // Keep only the last message
219 
220  // Create publisher for mocap to the pixhawk visual odometry topic
221  mocap_publisher_ = this->create_publisher<px4_msgs::msg::VehicleOdometry>(
222  "/fmu/in/vehicle_visual_odometry", qos_settings);
223 
224  try {
225 
226  m_udp_driver->init_receiver(m_ip, m_port);
227  m_udp_driver->receiver()->open();
228  m_udp_driver->receiver()->bind();
229  m_udp_driver->receiver()->asyncReceive(
230  std::bind(&UdpReceiverNode::receiver_callback, this, std::placeholders::_1));
231  } catch (const std::exception & ex) {
232  RCLCPP_ERROR(
233  get_logger(), "Error creating UDP receiver: %s:%i - %s",
234  m_ip.c_str(), m_port, ex.what());
235  return LNI::CallbackReturn::FAILURE;
236  }
237 
238  RCLCPP_DEBUG(get_logger(), "UDP receiver successfully configured.");
239 
240  return LNI::CallbackReturn::SUCCESS;
241 }

313 void UdpReceiverNode::receiver_callback(const std::vector<uint8_t> & buffer)
314 {
315  // Convert the buffer to a string
316  std::string buffer_str(buffer.begin(), buffer.end());
317 
318  // Check if the message format is valid
319  if (buffer_str.size() < 2 || buffer_str[0] != 'R' || buffer_str[1] != ',') {
320  FLIGHTSTACK_ERROR("Invalid message format!");
321  return; // Exit the function or handle the error accordingly
322  }
323 
324 
325  // Parse the data from the string
326  std::istringstream iss(buffer_str.substr(2)); // Skip "R," prefix
327  char comma;
328  iss >> mc_im.x >> comma >> mc_im.y >> comma >> mc_im.z >> comma // Read in the position
329  >> mc_im.q0 >> comma >> mc_im.q1 >> comma >> mc_im.q2 >> comma >> mc_im.q3 >> comma // Read in the quaternion
330  >> mc_im.vx >> comma >> mc_im.vy >> comma >> mc_im.vz >> comma // Read in the velocities
331  >> mc_im.rollspeed >> comma >> mc_im.pitchspeed >> comma >> mc_im.yawspeed; // Read in the angular speeds
332 
333  // Get the controller time
334  mc_im.control_time = vehicle_ptr->get_controltime();
335 
336  // Debug Mocap parsed data to terminal
337  // debugMocapData2screen();
338 
339  // Initilize the Mocap odometry message to publish to the pixhawk
340  px4_msgs::msg::VehicleOdometry out;
341 
342  // \brief: https://github.com/hpaul360/mocap_to_px4/blob/main/mocap_to_px4/converter_member_function.py
343  // Package the message
344  out.timestamp = 0; // Timestamp is automatically set inside PX4
345  out.timestamp_sample = 0; // Timestamp is automatically set inside PX4
346 
347  // Position frame is set to NED in the VICON code.
348  out.position[0] = (float) mc_im.x;
349  out.position[1] = (float) mc_im.y;
350  out.position[2] = (float) mc_im.z;
351 
352  // Orientation frame is set to NED in the VICON code.
353  out.q[0] = (float) mc_im.q0;
354  out.q[1] = (float) mc_im.q1;
355  out.q[2] = (float) mc_im.q2;
356  out.q[3] = (float) mc_im.q3;
357  out.pose_frame = POSE_FRAME_NED;
358 
359  // Velocity frame is set to NED in the VICON code.
360  // The velocity calculations in the VICON code is not great, once updated - fuse.
361  out.velocity[0] = (float) mc_im.vx;
362  out.velocity[1] = (float) mc_im.vy;
363  out.velocity[2] = (float) mc_im.vz;
364  out.velocity_frame = VELOCITY_FRAME_NED;
365 
366  // Angular speed is set to 0. We need to pass the velocites to EKF2 in the FRD Body-Fixed
367  // Frame. We do not have the data in the VICON code therefore we set it to 0.
368  out.angular_velocity[0] = (float) 0.0;
369  out.angular_velocity[1] = (float) 0.0;
370  out.angular_velocity[2] = (float) 0.0;
371 
372  // Other Parameters
373  out.position_variance = {0,0,0};
374  out.orientation_variance = {0,0,0};
375  out.velocity_variance = {0,0,0};
376 
377  // Publish the message
378  mocap_publisher_->publish(out);
379 
380  // Log the data
381  logLogData();
382 }
