67 
68  // Set the landing start time
69  // This will be sum of the start_time, the end of the flight_trajectory + the time needed to hover.
70  read.landing_start_time = read.start_time + read.traj_params.waypoint_times_.back() + read.hover_time;
71 
72  // Set the landing end time
73  // This will be the landing_start_time + 4.0 seconds.
74  read.landing_end_time = read.landing_start_time + 4.0;

78  read.end_time = read.landing_end_time + 1.0;

81  if (read.vio && read.mocap)

83  FLIGHTSTACK_ERROR("Both VIO and Mocap are active. Pick one fusion method. Terminating program.");
84  }
85  else if (!read.vio && !read.mocap)
