206 void mrac_pid::update(double time,
207  double x,
208  double y,
209  double z,
210  double vx,
211  double vy,
212  double vz,
213  double q0,
214  double q1,
215  double q2,
216  double q3,
217  double roll,
218  double pitch,
219  double yaw,
220  double w_x,
221  double w_y,
222  double w_z)
223 {
224  // 1. Assign all the states ----------------------------------------------------
225  cim.t = time;
226 
227  // Assign the Translational States
228  cim.x_tran.head<3>() << x,y,z;
229  cim.x_tran.tail<3>() << vx,vy,vz;
230 
231  // Assign the Rotational States
232  cim.eta_rot_q << roll,pitch,yaw;
233  cim.omega_rot_q << w_x, w_y, w_z;
234 
235  // Update the states to biplane mode. This function always measures the smallest angle.
236  // update_states_to_biplane_mode(cim.eta_rot_q, cim.omega_rot_q, cim.eta_rot_b, cim.omega_rot_b);
237 
238  // Assign the quaternions
239  cim.q.w() = q0;
240  cim.q.z() = q1;
241  cim.q.y() = q2;
242  cim.q.z() = q3;
243 
244  // Update the states to biplane mode. This function uses quaternions and tested code for gimbal locks
245  // Look at qrbp.hpp for more methods
246  // update_states_to_biplane_quaternions(cim.q, cim.eta_rot_q, cim.eta_rot_b, cim.omega_rot_b);
247  update_states_to_biplane_quaternion_gmk(cim.q, cim.omega_rot_q, cim.eta_rot_b, cim.omega_rot_b);
248 
249  // 2. Check if you are in biplane or Quadcopter mode ---------------------------
250  QRBPswitchFun(pitch, is_biplane, momentary_button);
251 
252  // 3. Get the reference trajectory ---------------------------------------------
253  ud.updateUserDefinedTrajectory(time);
254  cim.r_user = ud.getUserDefinedPosition();
255  cim.r_dot_user = ud.getUserDefinedVelocity();
256  cim.r_ddot_user = ud.getUserDefinedAcceleration();
257 
258  // 4. Capture the time before the execution of the controller ------------------
259  cim.alg_start_time = std::chrono::high_resolution_clock::now();
260 
261  // 5. Compute the rotation matrices
262  cim.R_Jq_I = rotationMatrix321LocalToGlobal(cim.eta_rot_q(0), cim.eta_rot_q(1), cim.eta_rot_q(2));
263  cim.R_I_Jq = rotationMatrix321GlobalToLocal(cim.eta_rot_q(0), cim.eta_rot_q(1), cim.eta_rot_q(2));
264  cim.R_Jb_I = rotationMatrix321LocalToGlobal(cim.eta_rot_b(0), cim.eta_rot_b(1), cim.eta_rot_b(2));
265  cim.R_I_Jb = rotationMatrix321GlobalToLocal(cim.eta_rot_b(0), cim.eta_rot_b(1), cim.eta_rot_b(2));
266 
267  // 6. Assign the rotation and the angular velocity for the inner loop here according to the mode
268  // and compute the Jacobin inverse and the angular rates. Also assign the inertia matrix
269  if (is_biplane)
270  {
271  cim.eta_rot = cim.eta_rot_b;
272  cim.omega_rot = cim.omega_rot_b;
273  cim.Jacobian = jacobianMatrix(cim.eta_rot_b(0), cim.eta_rot_b(1));
274  cim.Jacobian_inv = jacobianMatrixInverse(cim.eta_rot_b(0), cim.eta_rot_b(1));
275  cim.eta_rot_rate = cim.Jacobian_inv*cim.omega_rot_b;
276  cim.Jacobian_dot = jacobianMatrixDerivative(cim.eta_rot(0), cim.eta_rot(1),
277  cim.eta_rot_rate(0), cim.eta_rot_rate(1));
278  cip.inertia_matrix = inertia_matrix_b;
279  }
280  else
281  {
282  cim.eta_rot = cim.eta_rot_q;
283  cim.omega_rot = cim.omega_rot_q;
284  cim.Jacobian = jacobianMatrix(cim.eta_rot_q(0), cim.eta_rot_q(1));
285  cim.Jacobian_inv = jacobianMatrixInverse(cim.eta_rot_q(0), cim.eta_rot_q(1));
286  cim.eta_rot_rate = cim.Jacobian_inv*cim.omega_rot_q;
287  cim.Jacobian_dot = jacobianMatrixDerivative(cim.eta_rot(0), cim.eta_rot(1),
288  cim.eta_rot_rate(0), cim.eta_rot_rate(1));
289  cip.inertia_matrix = inertia_matrix_q;
290  }
291 
292  // 7. Compute the body velocities in the quadcopter frame from the inertial velocities
293  cim.x_tran_vel_Jq = cim.R_I_Jq * cim.x_tran.tail<3>();
294 
295  // 8. Compute the square of the norm of velocity in the body frame - it is the same for both quadcopter and biplane
296  // amd compute the aerodynamic angles
297  cim.aero.states = computeAeroStatesNACA0012Quad(cim.x_tran_vel_Jq(0), cim.x_tran_vel_Jq(1), cim.x_tran_vel_Jq(2),
298  cim.omega_rot_q(0), cim.omega_rot_q(1), cim.omega_rot_q(2));
299 
300  // 9. Compute the Direction Cosine Matrix in the quadcopter frame at the center of gravity
301  cim.R_W_Jq = aeroDCMQuadNED(cim.aero.states.alpha, cim.aero.states.beta);
302 
303  // 10. Compute the aerodynamic coefficeints
304  cim.aero.coeff = compute_aero_coefficients(cim.aero.states);
305 
306  // 11. Assign the values from the integrator -------------------------------------
307  // If the momentary button to indicate a switch is hit reset the integrator to
308  // emulate a start of a new mission in the inner loop or go ahead assigning the states as usual.
309  if (momentary_button){ reset_integrator_states(); } else{ assign_from_rk4(); }
310 
311 }

477 // Function to compute the outerloop control in I
478 void mrac_pid::compute_translational_control_in_I()
479 {
480  // Compute the error in the states
481  cim.e_tran << cim.x_tran - csm.x_tran_ref;
482  cim.e_tran_pos << cim.e_tran.head<3>();
483  cim.e_tran_vel << cim.e_tran.tail<3>();
484 
485  // Compute the translational position error between the reference model and the user defined trajectory
486  cim.e_tran_pos_ref << csm.x_tran_ref.head<3>() - cim.r_user;
487 
488  // Compute the reference command input [reference model - user_defined_trajectory]
489  cim.r_cmd_tran << MASS * (- cip.Ki_refmod_tran * csm.e_tran_pos_ref_I // Integral term
490  + cip.Kp_refmod_tran * cim.r_user // Proportional term
491  + cip.Kd_refmod_tran * cim.r_dot_user // Derivative term
492  + cim.r_ddot_user); // Feedforward term
493 
494  // Reference model
495  cim.x_tran_ref_dot << cip.A_ref_tran * csm.x_tran_ref
496  + cip.B_ref_tran * cim.r_cmd_tran;
497 
498  // Compute the baseline control input
499  cim.mu_tran_baseline << MASS * (- cip.Kp_tran * cim.e_tran_pos // Proportional term
500  - cip.Kd_tran * cim.e_tran_vel // Derivative term
501  - cip.Ki_tran * csm.e_tran_pos_I // Integral term
502  + cim.x_tran_ref_dot.tail<3>() // Feedforward term
503  - G * e3_basis) // Weight inversion term
504  - cim.aero.dyn.outer_loop_dynamic_inv; // Aerodynamic inversion term
505 
506  // Compute the augmented regressor vector
507  cim.augmented_outer_loop_regressor << cim.mu_tran_baseline,
508  cim.outer_loop_regressor;
509 
510  // Cache the transpose of the tracking error * P * B
511  Matrix<double, 1, 3> e_transpose_p_b = cim.e_tran.transpose() * cip.P_tran * cip.B_tran;
512 
513  // Adaptive laws
514  cim.K_hat_x_tran_dot << -cip.Gamma_x_tran *
515  cim.x_tran *
516  e_transpose_p_b;
517 
518  cim.K_hat_r_tran_dot << -cip.Gamma_r_tran *
519  cim.r_cmd_tran *
520  e_transpose_p_b;
521 
522  cim.Theta_hat_tran_dot << cip.Gamma_Theta_tran *
523  cim.augmented_outer_loop_regressor *
524  e_transpose_p_b;
525 
526  // Adaptive control law
527  cim.mu_tran_adaptive << csm.K_hat_x_tran.transpose() * cim.x_tran
528  + csm.K_hat_r_tran.transpose() * cim.r_cmd_tran
529  - csm.Theta_hat_tran.transpose() * cim.augmented_outer_loop_regressor;
530 
531  // Compute the total baseline + adaptive control
532  cim.mu_tran_I << cim.mu_tran_baseline + cim.mu_tran_adaptive;
533 }

535 // Compute the orientation commands and the desired total thrust
536 void mrac_pid::compute_u1_eta_d()
537 {
538  // Compute the virtual forces in the biplane frame
539  cim.mu_tran_Jb << cim.R_I_Jb * cim.mu_tran_I;
540 
541  // Compute the virutal forces in the quadcopter frame
542  cim.mu_tran_Jq << cim.R_I_Jq * cim.mu_tran_I;
543 
544  // compute u1 - Total Thrust in Biplane frame
545  double u1_b = sqrt( pow(cim.mu_tran_Jb(0), 2)
546  + pow(cim.mu_tran_Jb(1), 2)
547  + pow(cim.mu_tran_Jb(2), 2)
548  );
549 
550  // Compute u1 - Total Thrust in quadcopter frame
551  double u1_q = sqrt( pow(cim.mu_tran_Jq(0), 2)
552  + pow(cim.mu_tran_Jq(1), 2)
553  + pow(cim.mu_tran_Jq(2), 2)
554  );
555 
556  // Compute the desired roll in biplane frame
557  cim.eta_rot_d_b(0) = 0.0;
558 
559  // Compute the desired roll in the quadcopter frame
560  double r1 = cim.mu_tran_Jq(1) / u1_q;
561  cim.eta_rot_d_q(0) = atan2(r1, sqrt(1 - pow(r1,2)));
562 
563  // Compute the desired pitch in the biplane frame
564  cim.eta_rot_d_b(1) = atan2( cim.mu_tran_Jb(0), cim.mu_tran_Jb(2) );
565 
566  // Compute the desired pitch in the quadcopter frame
567  cim.eta_rot_d_q(1) = atan2( -cim.mu_tran_Jq(0), -cim.mu_tran_Jq(2) );
568 
569  // Compute the desired yaw in the biplane frame
570  double y1 = cim.mu_tran_Jb(1) / u1_b;
571  cim.eta_rot_d_b(2) = atan2(y1, sqrt(1 - pow(y1,2)));
572 
573  // Compute the desired yaw in quadcopter frame
574  cim.eta_rot_d_q(2) = ud.getUserDefinedYaw();
575 
576  // Compute the internal states - Roll - Quadcopter only
577  cim.internal_state_roll_d_filter_q << A_filter_roll_ref * csm.state_roll_d_filter_q
578  + B_filter_roll_ref * cim.eta_rot_d_q(0);
579 
580  cim.internal_state_roll_dot_d_filter_q << A_filter_roll_dot_ref * csm.state_roll_dot_d_filter_q
581  + B_filter_roll_dot_ref * cim.eta_rot_rate_d_q(0);
582 
583  // Compute the internal states - Pitch
584  cim.internal_state_pitch_d_filter_b << A_filter_pitch_ref * csm.state_pitch_d_filter_b
585  + B_filter_pitch_ref * cim.eta_rot_d_b(1);
586  cim.internal_state_pitch_d_filter_q << A_filter_pitch_ref * csm.state_pitch_d_filter_q
587  + B_filter_pitch_ref * cim.eta_rot_d_q(1);
588 
589  cim.internal_state_pitch_dot_d_filter_b << A_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter_b
590  + B_filter_pitch_dot_ref * cim.eta_rot_rate_d_b(1);
591  cim.internal_state_pitch_dot_d_filter_q << A_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter_q
592  + B_filter_pitch_dot_ref * cim.eta_rot_rate_d_q(1);
593 
594  // Compute the internal states - Yaw - Biplane only
595  cim.internal_state_yaw_d_filter_b << A_filter_yaw_ref * csm.state_yaw_d_filter_b
596  + B_filter_yaw_ref * cim.eta_rot_d_b(2);
597 
598  cim.internal_state_yaw_dot_d_filter_b << A_filter_yaw_dot_ref * csm.state_yaw_dot_d_filter_b
599  + B_filter_yaw_dot_ref * cim.eta_rot_rate_d_b(2);
600 
601  // Compute the desired Euler rates - Biplane
602  cim.eta_rot_rate_d_b(0) = 0.0;
603  cim.eta_rot_rate_d_b(1) = C_filter_pitch_ref * csm.state_pitch_d_filter_b;
604  cim.eta_rot_rate_d_b(2) = C_filter_yaw_ref * csm.state_yaw_d_filter_b;
605 
606  // Compute the desired Euler rates - Quadcopter
607  cim.eta_rot_rate_d_q(0) = C_filter_roll_ref * csm.state_roll_d_filter_q;
608  cim.eta_rot_rate_d_q(1) = C_filter_pitch_ref * csm.state_pitch_d_filter_q;
609  cim.eta_rot_rate_d_q(2) = ud.getUserDefinedYawRate();
610 
611  // Compute the desired Euler Acceleration - Biplane
612  cim.eta_rot_acceleration_d_b(0) = 0.0;
613  cim.eta_rot_acceleration_d_b(1) = C_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter_b;
614  cim.eta_rot_acceleration_d_b(2) = C_filter_yaw_dot_ref * csm.state_yaw_dot_d_filter_b;
615 
616  // Compute the desired Euler Acceleration - Quadcopter
617  cim.eta_rot_acceleration_d_q(0) = C_filter_roll_dot_ref * csm.state_roll_dot_d_filter_q;
618  cim.eta_rot_acceleration_d_q(1) = C_filter_pitch_dot_ref * csm.state_pitch_dot_d_filter_q;
619  cim.eta_rot_acceleration_d_q(2) = ud.getUserDefinedYawRateDot();
620 
621  // Assign according to the mode
622  if (is_biplane)
623  {
624  cim.u(0) = u1_b;
625  cim.mu_tran_J = cim.mu_tran_Jb;
626  cim.eta_rot_d = cim.eta_rot_d_b;
627  cim.eta_rot_rate_d = cim.eta_rot_rate_d_b;
628  cim.eta_rot_acceleration_d = cim.eta_rot_acceleration_d_b;
629  }
630  else
631  {
632  cim.u(0) = u1_q;
633  cim.mu_tran_J = cim.mu_tran_Jq;
634  cim.eta_rot_d = cim.eta_rot_d_q;
635  cim.eta_rot_rate_d = cim.eta_rot_rate_d_q;
636  cim.eta_rot_acceleration_d = cim.eta_rot_acceleration_d_q;
637  }

641 // Compute the rotational control
642 void mrac_pid::compute_rotational_control()
643 {
644  // If you are in biplane mode
645  if (is_biplane)
646  {
647  // Compute the error in the orientation [actual - desired]
648  cim.e_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
649  cim.e_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
650  cim.e_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
651 
652  // Compute the error in the angular velocity
653  cim.e_omega << cim.omega_rot - csm.omega_ref_rot;
654 
655  // Compute the error in the angular rates
656  cim.e_eta_rate << cim.eta_rot_rate - cim.eta_rot_rate_d;
657 
658  // Compute command angular velocity
659  cim.omega_cmd_rot << cim.Jacobian * (-cip.Kp_omega_cmd_rot_b * cim.e_eta
660  -cip.Ki_omega_cmd_rot_b * csm.e_eta_I
661  + cim.eta_rot_rate_d);
662 
663  // Compute the derivative of command angular velocity
664  cim.omega_cmd_dot_rot << cim.Jacobian_dot * (-cip.Kp_omega_cmd_rot_b * cim.e_eta
665  -cip.Ki_omega_cmd_rot_b * csm.e_eta_I
666  + cim.eta_rot_rate_d)
667  + cim.Jacobian * (-cip.Kp_omega_cmd_rot_b * cim.e_eta_rate
668  -cip.Ki_omega_cmd_rot_b * cim.e_eta
669  + cim.eta_rot_acceleration_d);
670 
671  // Compute the angular_velocity_error_ref [refmod - cmd]
672  cim.e_omega_ref << csm.omega_ref_rot - cim.omega_cmd_rot;
673 
674  // Reference command input
675  cim.r_cmd_rot << cip.Kp_omega_ref_rot_b * cim.omega_cmd_rot
676  -cip.Ki_omega_ref_rot_b * csm.e_omega_ref_I
677  +cim.omega_cmd_dot_rot;
678 
679  // Reference model
680  cim.omega_ref_dot_rot << -cip.Kp_omega_ref_rot_b * csm.omega_ref_rot + cim.r_cmd_rot;
681 
682  // Basline PID controller
683  cim.tau_rot_baseline << cip.inertia_matrix * ( -cip.Kp_rot_b * cim.e_eta // Proportional term
684  -cip.Kd_rot_b * cim.e_omega // Derivative term
685  -cip.Ki_rot_b * csm.e_eta_I // Integral term
686  + cim.omega_ref_dot_rot) // Feedforward term
687  + cim.omega_rot.cross(cip.inertia_matrix * cim.omega_rot) // Dynamic inversion term
688  - R_Jq_Jb*cim.aero.dyn.inner_loop_dynamic_inv; // Aero inversion term
689 
690  // Compute the augmented regressor vector
691  cim.augmented_inner_loop_regressor << cim.tau_rot_baseline, cim.inner_loop_regressor;
692 
693  // Cache the transpose of the tracking error * P * B
694  Eigen::Matrix<double, 1, 3> e_transpose_p_b = cim.e_omega.transpose() * cip.P_rot_b * cip.B_rot_b;
695 
696  // Adaptive laws
697  cim.K_hat_x_rot_dot << -cip.Gamma_x_rot_b *
698  cim.omega_rot *
699  e_transpose_p_b;
700 
701  cim.K_hat_r_rot_dot << -cip.Gamma_r_rot_b *
702  cim.r_cmd_rot *
703  e_transpose_p_b;
704 
705  cim.Theta_hat_rot_dot << cip.Gamma_Theta_rot_b *
706  cim.augmented_inner_loop_regressor *
707  e_transpose_p_b;
708 
709  // Adaptive control law
710  cim.tau_rot_adaptive << csm.K_hat_x_rot.transpose() * cim.omega_rot
711  +csm.K_hat_r_rot.transpose() * cim.r_cmd_rot
712  -csm.Theta_hat_rot.transpose() * cim.augmented_inner_loop_regressor;
713 
714  // Total rotational control input
715  cim.tau_rot << R_Jb_Jq * (cim.tau_rot_baseline + cim.tau_rot_adaptive);
716 
717 
718  }
719  // If you are in quadcopter mode
720  else
721  {
722  // Compute the error in the orientation [actual - desired]
723  cim.e_eta(0) = cim.eta_rot(0) - cim.eta_rot_d(0);
724  cim.e_eta(1) = cim.eta_rot(1) - cim.eta_rot_d(1);
725  cim.e_eta(2) = makeYawAngularErrorContinuous<double>(cim.eta_rot(2), cim.eta_rot_d(2));
726

728  cim.e_omega << cim.omega_rot - csm.omega_ref_rot;

736  + cim.eta_rot_rate_d);
737

740  -cip.Ki_omega_cmd_rot_q * csm.e_eta_I
