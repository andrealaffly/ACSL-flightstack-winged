151 bool pid_logger::logInitLogging() {
152  try {
153  // Create the directory path for the controller specific flight run logs
154  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "PID";
155 
156  // Check if the directory exits, and create it if it doesn't
157  if (!std::filesystem::exists(flight_run_controller_specific_directory))
158  {
159  std::filesystem::create_directories(flight_run_controller_specific_directory);
160  }
161 
162  // Generate the log file name with a timestamp
163  std::stringstream log_ss;
164  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
165  std::string log_filename = log_ss.str();
166 
167  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
168  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
169 
170  // Define a synchronous sink with a text ostream backend
171  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
172  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
173 
174  // Add a stream to the backend (in this case, a file stream)
175  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
176 
177  // Set the formatter for the sink
178  logdata_sink->set_formatter(
179  expr::stream
180  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
181  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
182  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
183  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
184  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
185  << expr::smessage // Log message
186  );
187 
188  // Add the sink to the logging core
189  logging::core::get()->add_sink(logdata_sink);
190 
191  // Set a filter for the sink
192  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
193 
194  logging::add_common_attributes(); // Add attributes like timestamp
195 
196  // Initilaize the logging headers
197  logInitHeaders();
198 
199  // Create the directory path for the controller specific flight run for parameters and gains used
200  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
201 
202  // Check if the direcotry exits, and create it if it doesn't
203  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
204  {
205  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
206  }
207 
208  // Copy the flight parameters over
209  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
210  std::stringstream params_target_ss;
211  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
212  std::string params_target_file = params_target_ss.str();
213 
214  if (!std::filesystem::exists(params_source_file)) {
215  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
216  }
217 
218  if (std::filesystem::exists(params_source_file)) {
219  std::filesystem::copy(params_source_file, params_target_file);
220  }
221 
222  // Copy the gains parameters over
223  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/qrbp/PID/gains_PID.json";
224  std::stringstream gains_target_ss;
225  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_PID.json" ;
226  std::string gains_target_file = gains_target_ss.str();
227 
228  if (!std::filesystem::exists(gains_source_file)) {
229  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
230  }
231 
232  if (std::filesystem::exists(gains_source_file)) {
233  std::filesystem::copy(gains_source_file, gains_target_file);
234  }
235 
236  // Return true if all successful
237  return true;
238 
239  } catch (const std::filesystem::filesystem_error& e) {
240  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
241  } catch (const std::exception& e) {
242  FLIGHTSTACK_ERROR("Exeption:", e.what());
243  }
244 
245  return false; // IF there are errors
246 }
