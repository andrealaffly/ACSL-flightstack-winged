141 bool pid_logger::logInitLogging() {
142  try {
143  // Create the directory path for the controller specific flight run logs
144  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "PID";
145 
146  // Check if the directory exits, and create it if it doesn't
147  if (!std::filesystem::exists(flight_run_controller_specific_directory))
148  {
149  std::filesystem::create_directories(flight_run_controller_specific_directory);
150  }
151 
152  // Generate the log file name with a timestamp
153  std::stringstream log_ss;
154  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
155  std::string log_filename = log_ss.str();
156 
157  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
158  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
159 
160  // Define a synchronous sink with a text ostream backend
161  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
162  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
163 
164  // Add a stream to the backend (in this case, a file stream)
165  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
166 
167  // Set the formatter for the sink
168  logdata_sink->set_formatter(
169  expr::stream
170  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
171  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
172  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
173  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
174  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
175  << expr::smessage // Log message
176  );
177 
178  // Add the sink to the logging core
179  logging::core::get()->add_sink(logdata_sink);
180 
181  // Set a filter for the sink
182  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
183 
184  logging::add_common_attributes(); // Add attributes like timestamp
185 
186  // Initilaize the logging headers
187  logInitHeaders();
188 
189  // Create the directory path for the controller specific flight run for parameters and gains used
190  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
191 
192  // Check if the direcotry exits, and create it if it doesn't
193  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
194  {
195  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
196  }
197 
198  // Copy the flight parameters over
199  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
200  std::stringstream params_target_ss;
201  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
202  std::string params_target_file = params_target_ss.str();
203 
204  if (!std::filesystem::exists(params_source_file)) {
205  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
206  }
207 
208  if (std::filesystem::exists(params_source_file)) {
209  std::filesystem::copy(params_source_file, params_target_file);
210  }
211 
212  // Copy the gains parameters over
213  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/qrbp/PID/gains_PID.json";
214  std::stringstream gains_target_ss;
215  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_PID.json" ;
216  std::string gains_target_file = gains_target_ss.str();
217 
218  if (!std::filesystem::exists(gains_source_file)) {
219  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
220  }
221 
222  if (std::filesystem::exists(gains_source_file)) {
223  std::filesystem::copy(gains_source_file, gains_target_file);
224  }
225 
226  // Return true if all successful
227  return true;
228 
229  } catch (const std::filesystem::filesystem_error& e) {
230  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
231  } catch (const std::exception& e) {
232  FLIGHTSTACK_ERROR("Exeption:", e.what());
233  }
234 
235  return false; // IF there are errors
236 }
