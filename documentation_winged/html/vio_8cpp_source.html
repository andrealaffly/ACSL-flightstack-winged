433 void VioT265Node::process_and_publish(const rs2::pose_frame & pose)
434 {
435  rs2_pose data = pose.get_pose_data();
436 
437  // Process the data
438  vio_im.control_time = vehicle_ptr->get_controltime();
439 
440  // ------------------------ Translational
441  // Translational data from the T265 is not rotated accourding to our mounting orientation. Therefore go from
442  // T265 frame to C, the frame analogous to the body frame (NED)
443  position_C << -data.translation.z,
444  data.translation.x,
445  -data.translation.y;
446 
447  // Translational data from the T265 is not rotated accourding to our mounting orientation. Therefore go from
448  // T265 frame to C, the frame analogous to the body frame (NED)
449  velocity_C << -data.velocity.z,
450  data.velocity.x,
451  -data.velocity.y;
452 
453  // Translational data from the T265 is not rotated accourding to our mounting orientation. Therefore go from
454  // T265 frame to C, the frame analogous to the body frame (NED)
455  acceleration_C << -data.acceleration.z,
456  data.acceleration.x,
457  -data.acceleration.y;
458 
459  // Assign the translational states.
460  vio_im.x = position_C(0);
461  vio_im.y = position_C(1);
462  vio_im.z = position_C(2);
463 
464  vio_im.vx = velocity_C(0);
465  vio_im.vy = velocity_C(1);
466  vio_im.vz = velocity_C(2);
467 
468  vio_im.ax = acceleration_C(0);
469  vio_im.ay = acceleration_C(1);
470  vio_im.az = acceleration_C(2);
471 
472  // ------------------------ Rotational
473  // Transform the quaternion from the camera frame (VR convention/pose_frame) to a frame c which is NED but at
474  // a specified orientation w.r.t the body frame (NED).
475  t265_quaternion = Quaternionf( data.rotation.w,
476  -data.rotation.z,
477  data.rotation.x,
478  -data.rotation.y);
479 
480  // Transform the orientation from the c frame to the frame C
481  // ---C frame is the analogous to the body frame (NED) and while the body frame is centered at the center of mass,
482  // this frame is shifted in x,y,z and positioned at center of the pose frame (VR frame) of the camera.
483  transformed_quaternion = (t265_quaternion*q_J_c_inv);
484 
485  // Final quaternion data in frame C w.r.t inertial frame I.
486  vio_im.q0 = transformed_quaternion.w();
487  vio_im.q1 = transformed_quaternion.x();
488  vio_im.q2 = transformed_quaternion.y();
489  vio_im.q3 = transformed_quaternion.z();
490 
491 
492  // Cache the angular velocity data with respect to the c frame that is mounted at a specified orientation w.r.t
493  // the body frame (NED)
494  angular_velocity_c << -data.angular_velocity.z,
495  data.angular_velocity.x,
496  -data.angular_velocity.y;
497 
498  // Cache the angular acceleration data with respect to the c frame that is mounted at a specified orientation w.r.t
499  // the body frame (NED)
500  angular_acceleration_c << -data.angular_acceleration.z,
501  data.angular_acceleration.x,
502  -data.angular_acceleration.y;
503 
504  // Rotate the angular velocity in frame c to frame C
505  // ---C frame is the analogous to the body frame (NED) and while the body frame is centered at the center of mass,
506  // this frame is shifted in x,y,z and positioned at center of the pose frame (VR frame) of the camera.
507  angular_velocity_C = R_c_C*angular_velocity_c;
508 
509  // Rotate the angular acceleration in frame c to frame C
510  // ---C frame is the analogous to the body frame (NED) and while the body frame is centered at the center of mass,
511  // this frame is shifted in x,y,z and positioned at center of the pose frame (VR frame) of the camera.
512  angular_acceleration_C = R_c_C*angular_acceleration_c;
513 
514  // Assign the rotational dynamics
515  vio_im.rollspeed = angular_velocity_C(0);
516  vio_im.pitchspeed = angular_velocity_C(1);
517  vio_im.yawspeed = angular_velocity_C(2);
518 
519  vio_im.rollacceleration = angular_velocity_C(0);
520  vio_im.pitchacceleration = angular_velocity_C(1);
521  vio_im.yawacceleration = angular_velocity_C(2);
522 
523  // Confidence Data
524  vio_im.tracker_conf = data.tracker_confidence;
525  vio_im.mapper_conf = data.mapper_confidence;
526 
527  // -----------------------> Publishing data to pixhawk
528  // Initialize the VIO odometry message to publish to the pixhawk
529  px4_msgs::msg::VehicleOdometry out;
530 
531  // \brief: https://github.com/hpaul360/mocap_to_px4/blob/main/mocap_to_px4/converter_member_function.py
532  // Pacakge the data to be sent out
533  out.timestamp = 0; // Timestamp is automatically set inside PX4
534  out.timestamp_sample = 0; // Timestamp is automatically set inside PX4
535 
536  // Position frame is set to NED in the code block above
537  out.position[0] = (float) vio_im.x;
538  out.position[1] = (float) vio_im.y;
539  out.position[2] = (float) vio_im.z;
540 
541  // Orientation frame is set to NED in the code block above
542  out.q[0] = (float) vio_im.q0;
543  out.q[1] = (float) vio_im.q1;
544  out.q[2] = (float) vio_im.q2;
545  out.q[3] = (float) vio_im.q3;
546  out.pose_frame = POSE_FRAME_NED;
547 
548  // Velocity frtame is set to NED in the code block above
549  // The velocity calculation in VIO is pretty good, so it will be fused.
550  out.velocity[0] = (float) vio_im.vx;
551  out.velocity[1] = (float) vio_im.vy;
552  out.velocity[2] = (float) vio_im.vz;
553  out.velocity_frame = VELOCITY_FRAME_NED;
554 
555  // Angular speed is passed on in VIO. We need to pass the velocites to EKF2 in the FRD Body-Fixed
556  // Frame. We have the code with VIO unlike vicon so we fuse the data.
557  out.angular_velocity[0] = (float) vio_im.rollspeed;
558  out.angular_velocity[1] = (float) vio_im.pitchspeed;
559  out.angular_velocity[2] = (float) vio_im.yawspeed;
560 
561  // Other Parameters
562  out.position_variance = {0,0,0};
563  out.orientation_variance = {0,0,0};
564  out.velocity_variance = {0,0,0};
565 
566  // Publish the message
567  vio_publisher_->publish(out);
568 
569  // Log the data
570  logLogData();
571 }
