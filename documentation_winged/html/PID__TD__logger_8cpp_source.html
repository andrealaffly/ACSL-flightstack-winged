137 bool pid_logger::logInitLogging() {
138  try {
139  // Create the directory path for the controller specific flight run logs
140  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "PID";
141 
142  // Check if the directory exits, and create it if it doesn't
143  if (!std::filesystem::exists(flight_run_controller_specific_directory))
144  {
145  std::filesystem::create_directories(flight_run_controller_specific_directory);
146  }
147 
148  // Generate the log file name with a timestamp
149  std::stringstream log_ss;
150  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
151  std::string log_filename = log_ss.str();
152 
153  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
154  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
155 
156  // Define a synchronous sink with a text ostream backend
157  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
158  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
159 
160  // Add a stream to the backend (in this case, a file stream)
161  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
162 
163  // Set the formatter for the sink
164  logdata_sink->set_formatter(
165  expr::stream
166  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
167  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
168  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
169  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
170  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
171  << expr::smessage // Log message
172  );
173 
174  // Add the sink to the logging core
175  logging::core::get()->add_sink(logdata_sink);
176 
177  // Set a filter for the sink
178  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
179 
180  logging::add_common_attributes(); // Add attributes like timestamp
181 
182  // Initilaize the logging headers
183  logInitHeaders();
184 
185  // Create the directory path for the controller specific flight run for parameters and gains used
186  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
187 
188  // Check if the direcotry exits, and create it if it doesn't
189  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
190  {
191  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
192  }
193 
194  // Copy the flight parameters over
195  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
196  std::stringstream params_target_ss;
197  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
198  std::string params_target_file = params_target_ss.str();
199 
200  if (!std::filesystem::exists(params_source_file)) {
201  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
202  }
203 
204  if (std::filesystem::exists(params_source_file)) {
205  std::filesystem::copy(params_source_file, params_target_file);
206  }
207 
208  // Copy the gains parameters over
209  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/rostestdrone/PID/gains_PID.json";
210  std::stringstream gains_target_ss;
211  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_PID.json" ;
212  std::string gains_target_file = gains_target_ss.str();
213 
214  if (!std::filesystem::exists(gains_source_file)) {
215  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
216  }
217 
218  if (std::filesystem::exists(gains_source_file)) {
219  std::filesystem::copy(gains_source_file, gains_target_file);
220  }
221 
222  // Return true if all successful
223  return true;
224 
225  } catch (const std::filesystem::filesystem_error& e) {
226  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
227  } catch (const std::exception& e) {
228  FLIGHTSTACK_ERROR("Exeption:", e.what());
229  }
230 
231  return false; // IF there are errors
232 }
