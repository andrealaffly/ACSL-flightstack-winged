145 bool pid_logger::logInitLogging() {
146  try {
147  // Create the directory path for the controller specific flight run logs
148  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "PID";
149 
150  // Check if the directory exits, and create it if it doesn't
151  if (!std::filesystem::exists(flight_run_controller_specific_directory))
152  {
153  std::filesystem::create_directories(flight_run_controller_specific_directory);
154  }
155 
156  // Generate the log file name with a timestamp
157  std::stringstream log_ss;
158  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
159  std::string log_filename = log_ss.str();
160 
161  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
162  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
163 
164  // Define a synchronous sink with a text ostream backend
165  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
166  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
167 
168  // Add a stream to the backend (in this case, a file stream)
169  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
170 
171  // Set the formatter for the sink
172  logdata_sink->set_formatter(
173  expr::stream
174  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
175  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
176  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
177  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
178  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
179  << expr::smessage // Log message
180  );
181 
182  // Add the sink to the logging core
183  logging::core::get()->add_sink(logdata_sink);
184 
185  // Set a filter for the sink
186  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
187 
188  logging::add_common_attributes(); // Add attributes like timestamp
189 
190  // Initilaize the logging headers
191  logInitHeaders();
192 
193  // Create the directory path for the controller specific flight run for parameters and gains used
194  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
195 
196  // Check if the direcotry exits, and create it if it doesn't
197  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
198  {
199  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
200  }
201 
202  // Copy the flight parameters over
203  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
204  std::stringstream params_target_ss;
205  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
206  std::string params_target_file = params_target_ss.str();
207 
208  if (!std::filesystem::exists(params_source_file)) {
209  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
210  }
211 
212  if (std::filesystem::exists(params_source_file)) {
213  std::filesystem::copy(params_source_file, params_target_file);
214  }
215 
216  // Copy the gains parameters over
217  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/rostestdrone/PID/gains_PID.json";
218  std::stringstream gains_target_ss;
219  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_PID.json" ;
220  std::string gains_target_file = gains_target_ss.str();
221 
222  if (!std::filesystem::exists(gains_source_file)) {
223  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
224  }
225 
226  if (std::filesystem::exists(gains_source_file)) {
227  std::filesystem::copy(gains_source_file, gains_target_file);
228  }
229 
230  // Return true if all successful
231  return true;
232 
233  } catch (const std::filesystem::filesystem_error& e) {
234  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
235  } catch (const std::exception& e) {
236  FLIGHTSTACK_ERROR("Exeption:", e.what());
237  }
238 
239  return false; // IF there are errors
240 }
