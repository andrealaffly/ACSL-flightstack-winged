237  void controlNode::run()
238  {
240  updateCurrentTime();
241 
242  // Debug to screen function
243  // debug2screen();
244 
245  // Main control loop state machine logic
246  switch (offboard_flag_)
247  {
248  case 0: // Logic before arm time
249  if (time_current_ > flight_params_ptr->arm_time) {
250  // Change to Offboard mode
251  this->publish_vehicle_command(px4_msgs::msg::VehicleCommand::VEHICLE_CMD_DO_SET_MODE,
252  ENABLE_CUSTOM_MODE, CUSTOM_MAIN_MODE_OFFBOARD);
253  // Arm the vehicle
254  this->arm();
255 
256  // Publish a control message to keep the pixhawk satisfied for a constant offboard signal
257  publish_offboard_control_mode();
258  publish_actuator_motors(get_zero_thrust(),
259  get_zero_thrust(),
260  get_zero_thrust(),
261  get_zero_thrust(),
262  get_zero_thrust(),
263  get_zero_thrust(),
264  get_zero_thrust(),
265  get_zero_thrust());
266 
267  // Change the state machine to state 1
268  offboard_flag_ = 1;
269  }
270  break;
271  // -------------------------------------------------------------
272  case 1: // Logic after arm time and before start time
273  if (time_current_ > flight_params_ptr->arm_time && time_current_ < flight_params_ptr->start_time)
274  {
275  // Publish least motor spin for quadcopters
276  publish_offboard_control_mode();
277 
278  if (PUBLISH_ACTUATOR)
279  {
280  publish_actuator_motors(get_min_thrust(),
281  get_min_thrust(),
282  get_min_thrust(),
283  get_min_thrust(),
284  get_zero_thrust(),
285  get_zero_thrust(),
286  get_zero_thrust(),
287  get_zero_thrust());
288  }
289  else
290  {
291  publish_actuator_motors(get_zero_thrust(),
292  get_zero_thrust(),
293  get_zero_thrust(),
294  get_zero_thrust(),
295  get_zero_thrust(),
296  get_zero_thrust(),
297  get_zero_thrust(),
298  get_zero_thrust());
299  }
300 
301  } else if (time_current_ > flight_params_ptr->start_time) {
302  // Change the state machine to state 2
303  offboard_flag_ = 2;
304  }
305  break;
306  // -------------------------------------------------------------
307  case 2: // Logic after start time
308  // Compute and publish the control command.
309  compute_and_publish_control();
310 
311  // If greater than end time.
312  if (time_current_ > flight_params_ptr->end_time)
313  {
314  // Change state machine to state 3
315  offboard_flag_ = 3;
316  }
317 
318  break;
319  // -------------------------------------------------------------
320  case 3: // Logic after end time
321 
322  // Publish zero thrust before disarming the vehicle
323  publish_actuator_motors(get_zero_thrust(),
324  get_zero_thrust(),
325  get_zero_thrust(),
326  get_zero_thrust(),
327  get_zero_thrust(),
328  get_zero_thrust(),
329  get_zero_thrust(),
330  get_zero_thrust());
331 
332  // Disarm the vehicle
333  this->disarm();
334 
335  // Reaffrim the offboard_flag_ to 3 and wait there
336  offboard_flag_ = 3;
337 
338  break;
339  // -------------------------------------------------------------
340  default:
341  // Error message for invalid controller choice
342  FLIGHTSTACK_ERROR("Error: Contol State Machine entered an invalid state. RUN!!!!!");
343  // Terminate the program
344  std::exit(EXIT_FAILURE);
345  break;
346  }
