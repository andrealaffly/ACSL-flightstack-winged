246  void controlNode::run()
247  {
249  updateCurrentTime();
250 
251  // Debug to screen function
252  // debug2screen();
253 
254  // Main control loop state machine logic
255  switch (offboard_flag_)
256  {
257  case 0: // Logic before arm time
258  if (time_current_ > flight_params_ptr->arm_time) {
259  // Change to Offboard mode
260  this->publish_vehicle_command(px4_msgs::msg::VehicleCommand::VEHICLE_CMD_DO_SET_MODE,
261  ENABLE_CUSTOM_MODE, CUSTOM_MAIN_MODE_OFFBOARD);
262  // Arm the vehicle
263  this->arm();
264 
265  // Publish a control message to keep the pixhawk satisfied for a constant offboard signal
266  publish_offboard_control_mode();
267  publish_actuator_motors(get_zero_thrust(),
268  get_zero_thrust(),
269  get_zero_thrust(),
270  get_zero_thrust(),
271  get_zero_thrust(),
272  get_zero_thrust(),
273  get_zero_thrust(),
274  get_zero_thrust());
275 
276  // Change the state machine to state 1
277  offboard_flag_ = 1;
278  }
279  break;
280  // -------------------------------------------------------------
281  case 1: // Logic after arm time and before start time
282  if (time_current_ > flight_params_ptr->arm_time && time_current_ < flight_params_ptr->start_time)
283  {
284  // Publish least motor spin for quadcopters
285  publish_offboard_control_mode();
286 
287  if (PUBLISH_ACTUATOR)
288  {
289  publish_actuator_motors(get_min_thrust(),
290  get_min_thrust(),
291  get_min_thrust(),
292  get_min_thrust(),
293  get_zero_thrust(),
294  get_zero_thrust(),
295  get_zero_thrust(),
296  get_zero_thrust());
297  }
298  else
299  {
300  publish_actuator_motors(get_zero_thrust(),
301  get_zero_thrust(),
302  get_zero_thrust(),
303  get_zero_thrust(),
304  get_zero_thrust(),
305  get_zero_thrust(),
306  get_zero_thrust(),
307  get_zero_thrust());
308  }
309 
310  } else if (time_current_ > flight_params_ptr->start_time) {
311  // Change the state machine to state 2
312  offboard_flag_ = 2;
313  }
314  break;
315  // -------------------------------------------------------------
316  case 2: // Logic after start time
317  // Compute and publish the control command.
318  compute_and_publish_control();
319 
320  // If greater than end time.
321  if (time_current_ > flight_params_ptr->end_time)
322  {
323  // Change state machine to state 3
324  offboard_flag_ = 3;
325  }
326 
327  break;
328  // -------------------------------------------------------------
329  case 3: // Logic after end time
330 
331  // Publish zero thrust before disarming the vehicle
332  publish_actuator_motors(get_zero_thrust(),
333  get_zero_thrust(),
334  get_zero_thrust(),
335  get_zero_thrust(),
336  get_zero_thrust(),
337  get_zero_thrust(),
338  get_zero_thrust(),
339  get_zero_thrust());
340 
341  // Disarm the vehicle
342  this->disarm();
343 
344  // Reaffrim the offboard_flag_ to 3 and wait there
345  offboard_flag_ = 3;
346 
347  break;
348  // -------------------------------------------------------------
349  default:
350  // Error message for invalid controller choice
351  FLIGHTSTACK_ERROR("Error: Contol State Machine entered an invalid state. RUN!!!!!");
352  // Terminate the program
353  std::exit(EXIT_FAILURE);
354  break;
355  }
356  }
