337 // Function to compute the outerloop control in I
338 void mrac_omega::compute_translational_control_in_I()
339 {
340  // Compute the error in the states
341  cim.e_tran << cim.x_tran - csm.x_tran_ref;
342  cim.e_tran_pos << cim.e_tran.head<3>();
343  cim.e_tran_vel << cim.e_tran.tail<3>();
344 
345  // Compute the translational position error between the reference model and the user defined trajectory
346  cim.e_tran_pos_ref << csm.x_tran_ref.head<3>() - cim.r_user;
347 
348  // Compute the reference command input [reference model - user_defined_trajectory]
349  cim.r_cmd_tran << MASS * (- cip.Ki_refmod_tran * csm.e_tran_pos_ref_I // Integral term
350  + cip.Kp_refmod_tran * cim.r_user // Proportional term
351  + cip.Kd_refmod_tran * cim.r_dot_user // Derivative term
352  + cim.r_ddot_user); // Feedforward term
353 
354  // Reference model
355  cim.x_tran_ref_dot << cip.A_ref_tran * csm.x_tran_ref
356  + cip.B_ref_tran * cim.r_cmd_tran;
357 
358  // Compute the baseline control input
359  cim.mu_tran_baseline << MASS * (- cip.Kp_tran * cim.e_tran_pos // Proportional term
360  - cip.Kd_tran * cim.e_tran_vel // Derivative term
361  - cip.Ki_tran * csm.e_tran_pos_I // Integral term
362  + cim.x_tran_ref_dot.tail<3>() // Feedforward term
363  - G * e3_basis) // Weight inversion term
364  - cim.aero.dyn.outer_loop_dynamic_inv; // Aerodynamic inverstion term
365 
366  // Compute the augemented regressor vector
367  cim.augmented_outer_loop_regressor << cim.mu_tran_baseline,
368  cim.outer_loop_regressor;
369 
370  // Cache the transpose of the tracking error * P * B
371  Matrix<double, 1, 3> e_transpose_p_b = cim.e_tran.transpose() * cip.P_tran * cip.B_tran;
372 
373  // Adaptive laws
374  cim.K_hat_x_tran_dot << -cip.Gamma_x_tran *
375  cim.x_tran *
376  e_transpose_p_b;
377 
378  cim.K_hat_r_tran_dot << -cip.Gamma_r_tran *
379  cim.r_cmd_tran *
380  e_transpose_p_b;
381 
382  cim.Theta_hat_tran_dot << cip.Gamma_Theta_tran *
383  cim.augmented_outer_loop_regressor *
384  e_transpose_p_b;
385 
386  // Adaptive control law
387  cim.mu_tran_adaptive << csm.K_hat_x_tran.transpose() * cim.x_tran
388  + csm.K_hat_r_tran.transpose() * cim.r_cmd_tran
389  - csm.Theta_hat_tran.transpose() * cim.augmented_outer_loop_regressor;
390 
391  // Compute the total baseline + adaptive control
392  cim.mu_tran_I << cim.mu_tran_baseline + cim.mu_tran_adaptive;
393

396 // Compute the orientation commands and the desired total thrust
397 void mrac_omega::compute_u1_eta_d()
398 {
399  // Compute the virtual forces in J
400  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
401 
402  // Compute u1 - Total Thrust
403  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
404  + pow(cim.mu_tran_J(1), 2)
405  + pow(cim.mu_tran_J(2), 2));
406 
407  // Compute the desired roll
408  double r1 = cim.mu_tran_J(1) / cim.u(0);
409  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
410 
411  // Compute the desired pitch
412  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
413 
414  // Compute the internal state for angular rates
415  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
416  + B_filter_roll_ref * cim.eta_rot_d(0);
417 
418  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
419  + B_filter_pitch_ref * cim.eta_rot_d(1);
420 
421 
422  // Compute the desired pitch and roll rates
423  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
424  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
425 
426  // Compute the Jacobian with the current orientation
427  cim.Jacobian = jacobianMatrix(cim.eta_rot(0), cim.eta_rot(1));
428 
429  // Compute the desired angular velocities
430  cim.omega_rot_d << cim.Jacobian * cim.eta_rot_rate_d;
431 
432  // Compute the internal state for angular accelration
433  cim.internal_state_omega_x_d_filter << A_filter_roll_dot_ref * csm.state_omega_x_d_filter
434  + B_filter_roll_dot_ref * cim.omega_rot_d(0);
435 
436  cim.internal_state_omega_y_d_filter << A_filter_pitch_dot_ref * csm.state_omega_y_d_filter
437  + B_filter_pitch_dot_ref * cim.omega_rot_d(1);
438 
439  cim.internal_state_omega_z_d_filter << A_filter_yaw_dot_ref * csm.state_omega_z_d_filter
440  + B_filter_yaw_dot_ref * cim.omega_rot_d(2);
441 
442  // Compute the desired angular acceleration
443  cim.alpha_rot_d(0) = C_filter_roll_dot_ref * csm.state_omega_x_d_filter;
444  cim.alpha_rot_d(1) = C_filter_pitch_dot_ref * csm.state_omega_y_d_filter;
445  cim.alpha_rot_d(2) = C_filter_yaw_dot_ref * csm.state_omega_z_d_filter;

463  cim.omega_cmd << cip.Kp_omega_ref * cim.omega_rot_d - cip.Ki_omega_ref * csm.e_rot_omega_ref_I + cim.alpha_rot_d;
464 
465  // Compute the derivative of the reference angular velocity
466  cim.omega_rot_ref_dot << cip.A_ref_rot * csm.omega_rot_ref + cip.B_ref_rot * cim.omega_cmd;
467 
468  // Baseline PID controller
469  cim.tau_rot_baseline << inertia_matrix_q * ( -cip.Kp_rot * cim.e_rot_eta // Proportional term
470  -cip.Kd_rot * cim.e_rot_omega // Derivative term
471  -cip.Ki_rot * csm.e_rot_eta_I // Integral term
472  + cim.omega_rot_ref_dot) // Feedforward term
473  + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot) // Dynamic inversion term
474  - cim.aero.dyn.inner_loop_dynamic_inv; // Aero inversion term
475 
476  // Compute the augmented regressor vector
477  cim.augmented_inner_loop_regressor << cim.tau_rot_baseline, cim.inner_loop_regressor;
478 
479  // Cache the tranpose of the tracking error * P * B
480  Eigen::Matrix<double, 1, 3> e_transpose_p_b = cim.e_rot_omega.transpose() * cip.P_rot * cip.B_rot;
481 
482  // Adaptive laws
483  cim.K_hat_x_rot_dot << -cip.Gamma_x_rot *
484  cim.omega_rot *
485  e_transpose_p_b;
486 
487  cim.K_hat_r_rot_dot << -cip.Gamma_r_rot *
488  cim.omega_cmd *
489  e_transpose_p_b;
490 
491  cim.Theta_hat_rot_dot << cip.Gamma_Theta_rot *
492  cim.augmented_inner_loop_regressor *
493  e_transpose_p_b;
494 
495  // Adaptive control law
496  cim.tau_rot_adaptive << csm.K_hat_x_rot.transpose() * cim.omega_rot
497  +csm.K_hat_r_rot.transpose() * cim.omega_cmd
498  -csm.Theta_hat_rot.transpose() * cim.augmented_inner_loop_regressor;
499 
500  // Total rotational control input
501  cim.tau_rot << cim.tau_rot_baseline + cim.tau_rot_adaptive;
502 
503 
504  // Assign the control inputs
505  cim.u(1) = cim.tau_rot(0);
506  cim.u(2) = cim.tau_rot(1);
507  cim.u(3) = cim.tau_rot(2);
508 }
