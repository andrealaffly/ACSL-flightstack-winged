326 // Function to compute the outerloop control in I
327 void mrac_omega::compute_translational_control_in_I()
328 {
329  // Compute the error in the states
330  cim.e_tran << cim.x_tran - csm.x_tran_ref;
331  cim.e_tran_pos << cim.e_tran.head<3>();
332  cim.e_tran_vel << cim.e_tran.tail<3>();
333 
334  // Compute the translational position error between the reference model and the user defined trajectory
335  cim.e_tran_pos_ref << csm.x_tran_ref.head<3>() - cim.r_user;
336 
337  // Compute the reference command input [reference model - user_defined_trajectory]
338  cim.r_cmd_tran << MASS * (- cip.Ki_refmod_tran * csm.e_tran_pos_ref_I // Integral term
339  + cip.Kp_refmod_tran * cim.r_user // Proportional term
340  + cip.Kd_refmod_tran * cim.r_dot_user // Derivative term
341  + cim.r_ddot_user); // Feedforward term
342 
343  // Reference model
344  cim.x_tran_ref_dot << cip.A_ref_tran * csm.x_tran_ref
345  + cip.B_ref_tran * cim.r_cmd_tran;
346 
347  // Compute the baseline control input
348  cim.mu_tran_baseline << MASS * (- cip.Kp_tran * cim.e_tran_pos // Proportional term
349  - cip.Kd_tran * cim.e_tran_vel // Derivative term
350  - cip.Ki_tran * csm.e_tran_pos_I // Integral term
351  + cim.x_tran_ref_dot.tail<3>() // Feedforward term
352  - G * e3_basis) // Weight inversion term
353  - cim.aero.dyn.outer_loop_dynamic_inv; // Aerodynamic inverstion term
354 
355  // Compute the augemented regressor vector
356  cim.augmented_outer_loop_regressor << cim.mu_tran_baseline,
357  cim.outer_loop_regressor;
358 
359  // Cache the transpose of the tracking error * P * B
360  Matrix<double, 1, 3> e_transpose_p_b = cim.e_tran.transpose() * cip.P_tran * cip.B_tran;
361 
362  // Adaptive laws
363  cim.K_hat_x_tran_dot << -cip.Gamma_x_tran *
364  cim.x_tran *
365  e_transpose_p_b;
366 
367  cim.K_hat_r_tran_dot << -cip.Gamma_r_tran *
368  cim.r_cmd_tran *
369  e_transpose_p_b;
370 
371  cim.Theta_hat_tran_dot << cip.Gamma_Theta_tran *
372  cim.augmented_outer_loop_regressor *
373  e_transpose_p_b;
374 
375  // Adaptive control law
376  cim.mu_tran_adaptive << csm.K_hat_x_tran.transpose() * cim.x_tran
377  + csm.K_hat_r_tran.transpose() * cim.r_cmd_tran
378  - csm.Theta_hat_tran.transpose() * cim.augmented_outer_loop_regressor;
379 
380  // Compute the total baseline + adaptive control
381  cim.mu_tran_I << cim.mu_tran_baseline + cim.mu_tran_adaptive;
382

385 // Compute the orientation commands and the desired total thrust
386 void mrac_omega::compute_u1_eta_d()
387 {
388  // Compute the virtual forces in J
389  cim.mu_tran_J << cim.R_I_J * cim.mu_tran_I;
390 
391  // Compute u1 - Total Thrust
392  cim.u(0) = sqrt( pow(cim.mu_tran_J(0), 2)
393  + pow(cim.mu_tran_J(1), 2)
394  + pow(cim.mu_tran_J(2), 2));
395 
396  // Compute the desired roll
397  double r1 = cim.mu_tran_J(1) / cim.u(0);
398  cim.eta_rot_d(0) = atan2(r1, sqrt(1 - pow(r1,2)));
399 
400  // Compute the desired pitch
401  cim.eta_rot_d(1) = atan2( -cim.mu_tran_J(0), -cim.mu_tran_J(2) );
402 
403  // Compute the internal state for angular rates
404  cim.internal_state_roll_d_filter << A_filter_roll_ref * csm.state_roll_d_filter
405  + B_filter_roll_ref * cim.eta_rot_d(0);
406 
407  cim.internal_state_pitch_d_filter << A_filter_pitch_ref * csm.state_pitch_d_filter
408  + B_filter_pitch_ref * cim.eta_rot_d(1);
409 
410 
411  // Compute the desired pitch and roll rates
412  cim.eta_rot_rate_d(0) = C_filter_roll_ref * csm.state_roll_d_filter;
413  cim.eta_rot_rate_d(1) = C_filter_pitch_ref * csm.state_pitch_d_filter;
414 
415  // Compute the Jacobian with the current orientation
416  cim.Jacobian = jacobianMatrix(cim.eta_rot(0), cim.eta_rot(1));
417 
418  // Compute the desired angular velocities
419  cim.omega_rot_d << cim.Jacobian * cim.eta_rot_rate_d;
420 
421  // Compute the internal state for angular accelration
422  cim.internal_state_omega_x_d_filter << A_filter_roll_dot_ref * csm.state_omega_x_d_filter
423  + B_filter_roll_dot_ref * cim.omega_rot_d(0);
424 
425  cim.internal_state_omega_y_d_filter << A_filter_pitch_dot_ref * csm.state_omega_y_d_filter
426  + B_filter_pitch_dot_ref * cim.omega_rot_d(1);
427 
428  cim.internal_state_omega_z_d_filter << A_filter_yaw_dot_ref * csm.state_omega_z_d_filter
429  + B_filter_yaw_dot_ref * cim.omega_rot_d(2);
430 
431  // Compute the desired angular acceleration
432  cim.alpha_rot_d(0) = C_filter_roll_dot_ref * csm.state_omega_x_d_filter;
433  cim.alpha_rot_d(1) = C_filter_pitch_dot_ref * csm.state_omega_y_d_filter;
434  cim.alpha_rot_d(2) = C_filter_yaw_dot_ref * csm.state_omega_z_d_filter;

452  cim.omega_cmd << cip.Kp_omega_ref * cim.omega_rot_d - cip.Ki_omega_ref * csm.e_rot_omega_ref_I + cim.alpha_rot_d;
453 
454  // Compute the derivative of the reference angular velocity
455  cim.omega_rot_ref_dot << cip.A_ref_rot * csm.omega_rot_ref + cip.B_ref_rot * cim.omega_cmd;
456 
457  // Baseline PID controller
458  cim.tau_rot_baseline << inertia_matrix_q * ( -cip.Kp_rot * cim.e_rot_eta // Proportional term
459  -cip.Kd_rot * cim.e_rot_omega // Derivative term
460  -cip.Ki_rot * csm.e_rot_eta_I // Integral term
461  + cim.omega_rot_ref_dot) // Feedforward term
462  + cim.omega_rot.cross(inertia_matrix_q * cim.omega_rot) // Dynamic inversion term
463  - cim.aero.dyn.inner_loop_dynamic_inv; // Aero inversion term
464 
465  // Compute the augmented regressor vector
466  cim.augmented_inner_loop_regressor << cim.tau_rot_baseline, cim.inner_loop_regressor;
467 
468  // Cache the tranpose of the tracking error * P * B
469  Eigen::Matrix<double, 1, 3> e_transpose_p_b = cim.e_rot_omega.transpose() * cip.P_rot * cip.B_rot;
470 
471  // Adaptive laws
472  cim.K_hat_x_rot_dot << -cip.Gamma_x_rot *
473  cim.omega_rot *
474  e_transpose_p_b;
475 
476  cim.K_hat_r_rot_dot << -cip.Gamma_r_rot *
477  cim.omega_cmd *
478  e_transpose_p_b;
479 
480  cim.Theta_hat_rot_dot << cip.Gamma_Theta_rot *
481  cim.augmented_inner_loop_regressor *
482  e_transpose_p_b;
483 
484  // Adaptive control law
485  cim.tau_rot_adaptive << csm.K_hat_x_rot.transpose() * cim.omega_rot
486  +csm.K_hat_r_rot.transpose() * cim.omega_cmd
487  -csm.Theta_hat_rot.transpose() * cim.augmented_inner_loop_regressor;
488 
489  // Total rotational control input
490  cim.tau_rot << cim.tau_rot_baseline + cim.tau_rot_adaptive;
491 
492 
493  // Assign the control inputs
494  cim.u(1) = cim.tau_rot(0);
495  cim.u(2) = cim.tau_rot(1);
496  cim.u(3) = cim.tau_rot(2);
497 }
