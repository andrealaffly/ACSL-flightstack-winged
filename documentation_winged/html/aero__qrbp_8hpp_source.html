116 // Inline function to compute the Aerodynamic angles for the NACA-0012 airfoil
128 inline AeroStates computeAeroStatesNACA0012Quad(const double u, const double v, const double w,
129  const double w_x, const double w_y, const double w_z)
130 {
131  AeroStates states;
132 
133  // Precompute common expressions
134  double u_sq = u * u;
135  double v_sq = v * v;
136  double w_sq = w * w;
137 
138  double LY_wz = LY * w_z;
139  double LY_wx = LY * w_x;
140 
141  // Compute intermediate values
142  double c2 = -(w - LX * w_y);
143  double c3 = -(w + LX * w_y);
144  double c4 = sqrt(u_sq + w_sq);
145  double c5 = sqrt((u + LY_wz) * (u + LY_wz) + (w - LY_wx) * (w - LY_wx));
146  double c6 = sqrt((u - LY_wz) * (u - LY_wz) + (w + LY_wx) * (w + LY_wx));
147 
148  double n1 = c4;
149  double n2 = sqrt(u_sq + c2 * c2);
150  double n3 = sqrt(u_sq + c3 * c3);
151  double n4 = sqrt(v * v + c4 * c4);
152  double n5 = sqrt(v * v + c5 * c5);
153  double n6 = sqrt(v * v + c6 * c6);
154 
155  // Compute the angle of attack at the center of mass
156  states.alpha = (n1 <= 0.01f) ? 0.0f : atan2(u, -w);
157 
158  // Compute the angle of attack of the upper wing
159  states.alpha_up = (n2 <= 0.01f) ? 0.0f : atan2(u, c2);
160 
161  // Compute the angle of attack of the lower wing
162  states.alpha_lw = (n3 <= 0.01f) ? 0.0f : atan2(u, c3);
163 
164  // Compute the side slip angle at the center of mass
165  states.beta = (n4 <= 0.01f) ? 0.0f : atan2(v, c4);
166 
167  // Compute the side slip angle for the left stabilizer
168  states.beta_lt = (n5 <= 0.01f) ? 0.0f : atan2(v, c5);
169 
170  // Compute the side slip angle for the right stabilizer
171  states.beta_rt = (n6 <= 0.01f) ? 0.0f : atan2(v, c6);
172 
173  // Compute the square of the norm of the velocity
174  states.v_norm_sq = u_sq + v_sq + w_sq;
175 
176  return states;
177 }

325 inline AeroDynamicMembers compute_aero_forces_moments(AeroStates states, AeroCoeff coeff, Eigen::Matrix3d R_J_I, Eigen::Matrix3d R_W_J)
326 {
327  AeroDynamicMembers output;
328 
329  // Cache the dynamic pressure of the wings and stabilizers
330  double dynamic_pressure_wings = DYN_PRESS_COEFF_W*states.v_norm_sq;
331  double dynamic_pressure_stabs = DYN_PRESS_COEFF_S*states.v_norm_sq;
332 
333  // Compute the Aerodynamic forces
334  output.F_aero(0) = -1*(dynamic_pressure_wings*(coeff.Cl_up + coeff.Cl_lw));
335  output.F_aero(1) = -1*(dynamic_pressure_wings*(coeff.Cd_up + coeff.Cd_lw)
336  + dynamic_pressure_stabs*(coeff.Cd_lt + coeff.Cd_rt));
337  output.F_aero(2) = -1*(dynamic_pressure_stabs*(coeff.Cl_lt + coeff.Cl_rt));
338 
339  // Assign it to dynamic inversion
340  output.outer_loop_dynamic_inv << R_J_I * R_W_J * output.F_aero;
341 
342  // cache the cosine and the sine values of alpha and beta at C.O.M
343  double calpha = cos(states.alpha);
344  double salpha = sin(states.alpha);
345  double cbeta = cos(states.beta);
346  double sbeta = sin(states.beta);
347 
348  // Compute the Aerodynamic moments
349  output.M_aero(0) = dynamic_pressure_wings*CHORD_W*(coeff.Cm_lt + coeff.Cm_rt)
350  - 2*LZ_S*(output.F_aero(1)*cbeta + output.F_aero(0)*sbeta);
351 
352  output.M_aero(1) = dynamic_pressure_wings*CHORD_W*(coeff.Cm_lw + coeff.Cm_up)
353  - 2*LZ_S*(output.F_aero(2)*calpha + output.F_aero(0)*cbeta*salpha
354  - output.F_aero(1)*salpha*sbeta);
355 
356  output.M_aero(2) = 0.0;
357 
358  // Assign it to the dynamic inversion
359  output.inner_loop_dynamic_inv = output.M_aero;
360 
361  // Return the output
362  return output;
363 
364 }
