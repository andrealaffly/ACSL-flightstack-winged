193  // Example usage : double result = makeYawAngularErrorContinuous<double>(yaw, user_defined_yaw);
194  // Example usage : float result = makeYawAngularErrorContinuous<float>(yaw_float, user_defined_yaw_float);
195  template <typename T>
206  inline T makeYawAngularErrorContinuous(T yaw, T user_defined_yaw)
207  {
208  T continuous_error;
209 
210  user_defined_yaw = wrapAngleToMinusPiAndPi(user_defined_yaw);
211 
212  T raw_error = yaw - user_defined_yaw;
213 
214  if (std::abs(raw_error) >= T(M_PI) && std::abs(raw_error) < T(2 * M_PI) && raw_error < T(0))
215  {
216  continuous_error = std::fmod(raw_error, T(M_PI)) + T(M_PI);
217  }
218  else if (std::abs(raw_error) >= T(M_PI) && std::abs(raw_error) < T(2 * M_PI) && raw_error > T(0))
219  {
220  continuous_error = std::fmod(raw_error, T(M_PI)) - T(M_PI);
221  }
222  else
223  {
224  continuous_error = std::fmod(raw_error, T(M_PI));
225  }
