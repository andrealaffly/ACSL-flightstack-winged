65  // Function for calculating the euler angles
66  // Computing Euler Angles from the unit quaternion following a 3-2-1 rotation sequence
67  // The Euler Angles are ordered as: roll, pitch, yaw
68  // For reference:
69  // - https://github.com/PX4/PX4-Matrix/blob/master/matrix/Euler.hpp
70  // - https://github.com/PX4/PX4-Matrix/blob/master/matrix/Dcm.hpp
71  std::array<double, 3> read_odometryNode::quaternionToEulerAnglesRPY(double q0, double q1, double q2, double q3)
72  {
73  std::array<double, 3> output;
74 
75  // Compute elements of Direction Cosine Matrix (DCM) from quaternion
76  double a = q0;
77  double b = q1;
78  double c = q2;
79  double d = q3;
80  double aa = a * a;
81  double ab = a * b;
82  double ac = a * c;
83  double ad = a * d;
84  double bb = b * b;
85  double bc = b * c;
86  double bd = b * d;
87  double cc = c * c;
88  double cd = c * d;
89  double dd = d * d;
90 
91  // Compute DCM elements
92  double dcm00 = aa + bb - cc - dd;
93  // double dcm01 = 2 * (bc - ad); // not needed by the algorithm
94  double dcm02 = 2 * (ac + bd);
95  double dcm10 = 2 * (bc + ad);
96  // double dcm11 = aa - bb + cc - dd; // not needed by the algorithm
97  double dcm12 = 2 * (cd - ab);
98  double dcm20 = 2 * (bd - ac);
99  double dcm21 = 2 * (ab + cd);
100  double dcm22 = aa - bb - cc + dd;
101 
102  // Clamp the value of dcm20 to the range [-1, 1] to avoid domain errors with asinf
103  if (dcm20 < -1.0) dcm20 = -1.0;
104  if (dcm20 > 1.0) dcm20 = 1.0;
105 
106  // Compute the Euler angles from DCM
107  output[1] = asin(-dcm20); // Pitch
108 
109  // Handle gimbal lock cases
110  if (fabs(output[1] - M_PI / 2) < 1.0e-3) {
111  output[0] = 0.0; // Roll
112  output[2] = atan2(dcm12, dcm02); // Yaw
113  } else if (fabs(output[1] + M_PI / 2) < 1.0e-3) {
114  output[0] = 0.0; // Roll
115  output[2] = atan2(-dcm12, -dcm02); // Yaw
116  } else {
117  output[0] = atan2(dcm21, dcm22); // Roll
118  output[2] = atan2(dcm10, dcm00); // Yaw
119  }
120 
121 
122  return output;
123
