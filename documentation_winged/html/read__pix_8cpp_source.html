71  // Function for calculating the euler angles
72  // Computing Euler Angles from the unit quaternion following a 3-2-1 rotation sequence
73  // The Euler Angles are ordered as: roll, pitch, yaw
74  // For reference:
75  // - https://github.com/PX4/PX4-Matrix/blob/master/matrix/Euler.hpp
76  // - https://github.com/PX4/PX4-Matrix/blob/master/matrix/Dcm.hpp
77  std::array<double, 3> read_odometryNode::quaternionToEulerAnglesRPY(double q0, double q1, double q2, double q3)
78  {
79  std::array<double, 3> output;
80 
81  // Compute elements of Direction Cosine Matrix (DCM) from quaternion
82  double a = q0;
83  double b = q1;
84  double c = q2;
85  double d = q3;
86  double aa = a * a;
87  double ab = a * b;
88  double ac = a * c;
89  double ad = a * d;
90  double bb = b * b;
91  double bc = b * c;
92  double bd = b * d;
93  double cc = c * c;
94  double cd = c * d;
95  double dd = d * d;
96 
97  // Compute DCM elements
98  double dcm00 = aa + bb - cc - dd;
99  // double dcm01 = 2 * (bc - ad); // not needed by the algorithm
100  double dcm02 = 2 * (ac + bd);
101  double dcm10 = 2 * (bc + ad);
102  // double dcm11 = aa - bb + cc - dd; // not needed by the algorithm
103  double dcm12 = 2 * (cd - ab);
104  double dcm20 = 2 * (bd - ac);
105  double dcm21 = 2 * (ab + cd);
106  double dcm22 = aa - bb - cc + dd;
107 
108  // Clamp the value of dcm20 to the range [-1, 1] to avoid domain errors with asinf
109  if (dcm20 < -1.0) dcm20 = -1.0;
110  if (dcm20 > 1.0) dcm20 = 1.0;
111 
112  // Compute the Euler angles from DCM
113  output[1] = asin(-dcm20); // Pitch
114 
115  // Handle gimbal lock cases
116  if (fabs(output[1] - M_PI / 2) < 1.0e-3) {
117  output[0] = 0.0; // Roll
118  output[2] = atan2(dcm12, dcm02); // Yaw
119  } else if (fabs(output[1] + M_PI / 2) < 1.0e-3) {
120  output[0] = 0.0; // Roll
121  output[2] = atan2(-dcm12, -dcm02); // Yaw
122  } else {
123  output[0] = atan2(dcm21, dcm22); // Roll
124  output[2] = atan2(dcm10, dcm00); // Yaw
125  }
126 
127 
128  return output;
129 
130  }
