217 bool mrac_pid_logger::logInitLogging() {
218  try {
219  // Create the directory path for the controller specific flight run logs
220  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "MRAC_PID";
221 
222  // Check if the directory exits, and create it if it doesn't
223  if (!std::filesystem::exists(flight_run_controller_specific_directory))
224  {
225  std::filesystem::create_directories(flight_run_controller_specific_directory);
226  }
227 
228  // Generate the log file name with a timestamp
229  std::stringstream log_ss;
230  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
231  std::string log_filename = log_ss.str();
232 
233  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
234  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
235 
236  // Define a synchronous sink with a text ostream backend
237  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
238  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
239 
240  // Add a stream to the backend (in this case, a file stream)
241  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
242 
243  // Set the formatter for the sink
244  logdata_sink->set_formatter(
245  expr::stream
246  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
247  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
248  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
249  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
250  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
251  << expr::smessage // Log message
252  );
253 
254  // Add the sink to the logging core
255  logging::core::get()->add_sink(logdata_sink);
256 
257  // Set a filter for the sink
258  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
259 
260  logging::add_common_attributes(); // Add attributes like timestamp
261 
262  // Initilaize the logging headers
263  logInitHeaders();
264 
265  // Create the directory path for the controller specific flight run for parameters and gains used
266  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
267 
268  // Check if the direcotry exits, and create it if it doesn't
269  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
270  {
271  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
272  }
273 
274  // Copy the flight parameters over
275  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
276  std::stringstream params_target_ss;
277  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
278  std::string params_target_file = params_target_ss.str();
279 
280  if (!std::filesystem::exists(params_source_file)) {
281  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
282  }
283 
284  if (std::filesystem::exists(params_source_file)) {
285  std::filesystem::copy(params_source_file, params_target_file);
286  }
287 
288  // Copy the gains parameters over
289  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/qrbp/MRAC_PID/gains_MRAC_PID.json";
290  std::stringstream gains_target_ss;
291  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_MRAC_PID.json" ;
292  std::string gains_target_file = gains_target_ss.str();
293 
294  if (!std::filesystem::exists(gains_source_file)) {
295  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
296  }
297 
298  if (std::filesystem::exists(gains_source_file)) {
299  std::filesystem::copy(gains_source_file, gains_target_file);
300  }
301 
302  // Return true if all successful
303  return true;
304 
305  } catch (const std::filesystem::filesystem_error& e) {
306  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
307  } catch (const std::exception& e) {
308  FLIGHTSTACK_ERROR("Exeption:", e.what());
309  }
310 
311  return false; // IF there are errors
312 }
