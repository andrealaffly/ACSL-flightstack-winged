227 bool mrac_omega_logger::logInitLogging() {
228  try {
229  // Create the directory path for the controller specific flight run logs
230  std::string flight_run_controller_specific_directory = flight_run_log_directory + "/" + "MRAC_OMEGA";
231 
232  // Check if the directory exits, and create it if it doesn't
233  if (!std::filesystem::exists(flight_run_controller_specific_directory))
234  {
235  std::filesystem::create_directories(flight_run_controller_specific_directory);
236  }
237 
238  // Generate the log file name with a timestamp
239  std::stringstream log_ss;
240  log_ss << flight_run_controller_specific_directory << "/controller_log" << ".log";
241  std::string log_filename = log_ss.str();
242 
243  // Add the "Tag" attribute with a constant value of "LogDataTag" to the logger
244  logger_logdata.add_attribute("Tag", attrs::constant<std::string>("LogDataTag"));
245 
246  // Define a synchronous sink with a text ostream backend
247  typedef sinks::synchronous_sink<sinks::text_ostream_backend> text_sink;
248  boost::shared_ptr<text_sink> logdata_sink = boost::make_shared<text_sink>();
249 
250  // Add a stream to the backend (in this case, a file stream)
251  logdata_sink->locked_backend()->add_stream(boost::make_shared<std::ofstream>(log_filename));
252 
253  // Set the formatter for the sink
254  logdata_sink->set_formatter(
255  expr::stream
256  << "[" << expr::format_date_time<boost::posix_time::ptime>("TimeStamp", "%Y-%m-%d %H:%M:%S.%f") << "] " // Format date and time
257  << "[" << expr::attr<boost::log::attributes::current_thread_id::value_type>("ThreadID") << "] " // Current thread ID
258  << "[" << expr::attr<std::string>("Tag") << "] " // Tag attribute value
259  << "[" << expr::attr<boost::log::attributes::current_process_id::value_type>("ProcessID") << "] " // Current process ID
260  << "[" << expr::attr<unsigned int>("LineID") << "] " // Line ID
261  << expr::smessage // Log message
262  );
263 
264  // Add the sink to the logging core
265  logging::core::get()->add_sink(logdata_sink);
266 
267  // Set a filter for the sink
268  logdata_sink->set_filter(expr::has_attr("Tag") && expr::attr<std::string>("Tag") == "LogDataTag");
269 
270  logging::add_common_attributes(); // Add attributes like timestamp
271 
272  // Initilaize the logging headers
273  logInitHeaders();
274 
275  // Create the directory path for the controller specific flight run for parameters and gains used
276  std::string flight_run_controller_specific_parameter_directory = flight_run_controller_specific_directory + "/" + "params";
277 
278  // Check if the direcotry exits, and create it if it doesn't
279  if (!std::filesystem::exists(flight_run_controller_specific_parameter_directory))
280  {
281  std::filesystem::create_directories(flight_run_controller_specific_parameter_directory);
282  }
283 
284  // Copy the flight parameters over
285  std::string params_source_file = "./src/acsl_flight/params/flight_params/flight_main_params.json";
286  std::stringstream params_target_ss;
287  params_target_ss << flight_run_controller_specific_parameter_directory << "/flight_main_params.json" ;
288  std::string params_target_file = params_target_ss.str();
289 
290  if (!std::filesystem::exists(params_source_file)) {
291  FLIGHTSTACK_ERROR("MAIN PARAMS FILE NOT PRESENT");
292  }
293 
294  if (std::filesystem::exists(params_source_file)) {
295  std::filesystem::copy(params_source_file, params_target_file);
296  }
297 
298  // Copy the gains parameters over
299  std::string gains_source_file = "./src/acsl_flight/params/control_algorithms/qrbp/MRAC_OMEGA/gains_MRAC_OMEGA.json";
300  std::stringstream gains_target_ss;
301  gains_target_ss << flight_run_controller_specific_parameter_directory << "/gains_MRAC_OMEGA.json" ;
302  std::string gains_target_file = gains_target_ss.str();
303 
304  if (!std::filesystem::exists(gains_source_file)) {
305  FLIGHTSTACK_ERROR("GAINS FILE NOT PRESENT");
306  }
307 
308  if (std::filesystem::exists(gains_source_file)) {
309  std::filesystem::copy(gains_source_file, gains_target_file);
310  }
311 
312  // Return true if all successful
313  return true;
314 
315  } catch (const std::filesystem::filesystem_error& e) {
316  FLIGHTSTACK_ERROR("Filesystem error:", e.what());
317  } catch (const std::exception& e) {
318  FLIGHTSTACK_ERROR("Exception:", e.what());
319  }
320 
321  return false; // IF there are errors
322 }
