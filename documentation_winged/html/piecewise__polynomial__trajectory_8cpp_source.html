68 std::vector<Eigen::MatrixXd> piecewise_polynomial_trajectory::polyCoefAssigning
69 (const std::vector<std::vector<double>>& poly_coeff_matrix_in)
70 {
71  // Get the size of the input matrix
72  int rows = poly_coeff_matrix_in.size();
73  int cols = poly_coeff_matrix_in[0].size();
74 
75  // Calculate the size of the output matrices
76  int out_rows = rows / 3;
77 
78  // Initialize output matrices for x, y, and z variables
79  Eigen::MatrixXd poly_coeff_matrix_out_x(out_rows, cols);
80  Eigen::MatrixXd poly_coeff_matrix_out_y(out_rows, cols);
81  Eigen::MatrixXd poly_coeff_matrix_out_z(out_rows, cols);
82 
83  // Assign polynomial coefficients to designated variables
84  for (int i = 0; i < out_rows; ++i) {
85  for (int j = 0; j < cols; ++j) {
86  poly_coeff_matrix_out_x(i, j) = poly_coeff_matrix_in[3 * i][j];
87  poly_coeff_matrix_out_y(i, j) = poly_coeff_matrix_in[3 * i + 1][j];
88  poly_coeff_matrix_out_z(i, j) = poly_coeff_matrix_in[3 * i + 2][j];
89  }
90  }
91 
92  // Return the output matrices
93  return {poly_coeff_matrix_out_x, poly_coeff_matrix_out_y, poly_coeff_matrix_out_z};
94 }

152 // Function that updates the user-defined trajectory based on the time at which it is requested to be evaluated
153 void piecewise_polynomial_trajectory::updateUserDefinedTrajectory(double controller_time)
154 {
155  // Adjust the current controller time so that the piecewise polynomial trajectory starts at 0s.
156  // The coefffiecnts are generated in matlab assuming we start at 0.
157  internal_time = controller_time - flight_params_ptr->start_time;
158 
159  // Adjust time and identify segment
160  // NOTE: If the current time has passed the last waypoint time, then segment_ = 0;
161  std::tie(internal_time_adjusted, segment_) = polynomialTimeAdjusted(internal_time);
162 
163  // If the internal time has passed the last waypoint time but it is lesser than the landing start time,
164  // Set the trajectory to perform HOVER in the last position of the trajectory
165  if(internal_time >= flight_params_ptr->traj_params.waypoint_times_.back() &&
166  controller_time < flight_params_ptr->landing_start_time)
167  {
168  setUserDefinedPosition(user_defined_position_previous_);
169  setUserDefinedVelocity(Vector3d::Zero());
170  setUserDefinedAcceleration(Vector3d::Zero());
171  }
172  // If the controller_time has passed the landing start time but not the landing end time,
173  // perform the LANDING maneuver
174  else if (controller_time >= flight_params_ptr->landing_start_time &&
175  controller_time < flight_params_ptr->landing_end_time)
176  {
177  setUserDefinedPosition(Vector3d(user_defined_position_previous_(0),
178  user_defined_position_previous_(1),
179  evaluatePolynomial(landing_position_coef_z_,
180  controller_time - flight_params_ptr->landing_start_time )));
181 
182  // Evaluate and update user_defined_velocity_
183  setUserDefinedVelocity(Vector3d(0.0,
184  0.0,
185  evaluatePolynomial(landing_velocity_coef_z_,
186  controller_time - flight_params_ptr->landing_start_time )));
187 
188  // Evaluate and update user_defined_acceleration_
189  setUserDefinedAcceleration(Vector3d(0.0,
190  0.0,
191  evaluatePolynomial(landing_velocity_coef_z_,
192  controller_time - flight_params_ptr->landing_start_time )));
193  }
194  // If the current time has passed the landing end time, stay at the origin
195  else if (controller_time >= flight_params_ptr->landing_end_time)
196  {
197  setUserDefinedPosition(user_defined_position_previous_);
198  setUserDefinedVelocity(Vector3d::Zero());
199  setUserDefinedAcceleration(Vector3d::Zero());
200  }
201  // If any of the above conditions are not met, follow the user defined trajectory
202  else
203  {
204  // Evaluate and update user_defined_position_
205  setUserDefinedPosition(Vector3d(
206  evaluatePolynomial(position_coef_x_.row(segment_), internal_time_adjusted),
207  evaluatePolynomial(position_coef_y_.row(segment_), internal_time_adjusted),
208  evaluatePolynomial(position_coef_z_.row(segment_), internal_time_adjusted)));
209 
210  // Evaluate and update user_defined_velocity_
211  setUserDefinedVelocity(Vector3d(
212  evaluatePolynomial(velocity_coef_x_.row(segment_), internal_time_adjusted),
213  evaluatePolynomial(velocity_coef_y_.row(segment_), internal_time_adjusted),
214  evaluatePolynomial(velocity_coef_z_.row(segment_), internal_time_adjusted)));
215 
216  // Evaluate and update user_defined_acceleration_
217  setUserDefinedAcceleration(Vector3d(
218  evaluatePolynomial(acceleration_coef_x_.row(segment_), internal_time_adjusted),
219  evaluatePolynomial(acceleration_coef_y_.row(segment_), internal_time_adjusted),
220  evaluatePolynomial(acceleration_coef_z_.row(segment_), internal_time_adjusted)));
221  }
222 
223  // Update user_defined_position_previous_
224  user_defined_position_previous_ = getUserDefinedPosition();
225 
226  // Update the user defined yaw
227  updateUserDefinedYaw();
228 }

231 // to be evaluated
232 void piecewise_polynomial_trajectory::updateUserDefinedYaw()
233 {
234  // Compute the 2D norm of the trajectory velocity in the XY plane
235  norm_velocity_XY_ = norm2D<Eigen::VectorXd, double>(velocity_coef_x_.row(segment_),
236  velocity_coef_y_.row(segment_),
237  internal_time_adjusted);
238 
239  // If the norm of the velocity in the XY plane is close to zero OR If the current time has passed the

241  if (this->norm_velocity_XY_ < 1e-3 || (internal_time >= flight_params_ptr->traj_params.waypoint_times_.back()))
242  {
243  setUserDefinedYaw(user_defined_yaw_previous_);
244  setUserDefinedYawRate(0.0);
245  setUserDefinedYawRateDot(0.0);
246  }
247  else
248  {
249  // Update the User defined yaw angle

277 double piecewise_polynomial_trajectory::yawComputation(const VectorXd& Vx_coef, const VectorXd& Vy_coef, double t)
278 {

289 // Compute the user-defined yaw_dot from the trajectory velocity in the XY plane
290 double piecewise_polynomial_trajectory:: yawDotComputation(const VectorXd& Vx_coef, const VectorXd& Vy_coef,
291  const VectorXd& Ax_coef, const VectorXd& Ay_coef,
292  double t)
293 {
294  // Compute the acceleration components Ax and Ay at time t using polynomial coefficients
295  double Ax = evaluatePolynomial(Ax_coef, t);
296  double Ay = evaluatePolynomial(Ay_coef, t);
297 
298  // Compute the acceleration angle in the complex plane
299  double acceleration_angle = std::atan2(Ay, Ax);
300 
301  // Compute the velocity and acceleration norms
302  double velocity_norm = norm2D<Eigen::VectorXd, double>(Vx_coef, Vy_coef, t);
303  double acceleration_norm = norm2D<Eigen::VectorXd, double>(Ax_coef, Ay_coef, t);
304 
305  // Compute the yaw angle
306  double yaw = yawComputation(Vx_coef, Vy_coef, t);
307 
308  // Compute the yaw rate
309  double yaw_dot = (acceleration_norm / velocity_norm) * std::sin(acceleration_angle - yaw);
310 
311  return yaw_dot;
312 }

314 // Compute the user-defined yaw_dot_dot from the trajectory velocity in the XY plane
315 double piecewise_polynomial_trajectory::yawDotDotComputation(const VectorXd& Vx_coef, const VectorXd& Vy_coef,
316  const VectorXd& Ax_coef, const VectorXd& Ay_coef,
317  const VectorXd& Jx_coef, const VectorXd& Jy_coef,
318  double t)
319 {
320  // Compute the jerk components Jx and Jy at time t using polynomial coefficients
321  double Jx = evaluatePolynomial(Jx_coef, t);
322  double Jy = evaluatePolynomial(Jy_coef, t);
323 
324  // Compute the jerk angle in the complex plane
325  double jerk_angle = std::atan2(Jy, Jx);
326 
327  // Compute the velocity and jerk norms
328  double velocity_norm = norm2D<Eigen::VectorXd, double>(Vx_coef, Vy_coef, t);
329  double jerk_norm = norm2D<Eigen::VectorXd, double>(Jx_coef, Jy_coef, t);
330 
331  // Compute the derivative of the velocity norm
332  double velocity_norm_prime = norm2Dderivative<Eigen::VectorXd, double>(Vx_coef, Vy_coef, Ax_coef, Ay_coef, t);
333 
334  // Compute the yaw angle and its derivative
335  double yaw = yawComputation(Vx_coef, Vy_coef, t);
336  double yaw_dot = yawDotComputation(Vx_coef, Vy_coef, Ax_coef, Ay_coef, t);
337 
338  // Compute the second derivative of yaw angle
339  double yaw_dot_dot = (jerk_norm * std::sin(jerk_angle - yaw) -
340  2 * velocity_norm_prime * yaw_dot) / velocity_norm;
341 
342  return yaw_dot_dot;
343 }
