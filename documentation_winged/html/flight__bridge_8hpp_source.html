154 {
155  bool success = true;
156  sched_param params;
157  int policy;
158  success &= (pthread_getschedparam(native_handle, &policy, &params) == 0);
159  if (priority == ThreadPriority::HIGH) {
160  params.sched_priority = sched_get_priority_max(SCHED_FIFO);
161  } else {
162  params.sched_priority = sched_get_priority_min(SCHED_FIFO);
163  }
164  success &= (pthread_setschedparam(native_handle, SCHED_FIFO, &params) == 0);
165  if (cpu_id >= 0) {
166  cpu_set_t cpuset;
167  CPU_ZERO(&cpuset);
168  CPU_SET(cpu_id, &cpuset);
169  success &= (pthread_setaffinity_np(native_handle, sizeof(cpu_set_t), &cpuset) == 0);
170  }
171 
172  return success;
173 }
